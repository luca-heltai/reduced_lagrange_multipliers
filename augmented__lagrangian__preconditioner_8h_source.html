53 {
54 v.block(0) = 0.;
55 v.block(1) = 0.;
56

90 Epetra_CrsMatrix A_trilinos = A.trilinos_matrix();
91 Epetra_CrsMatrix Ct_trilinos = Ct.trilinos_matrix();
92 auto multi_vector = scaling_vector.trilinos_vector();
93
94
95 Assert((A_trilinos.NumGlobalRows() !=
96 Ct_trilinos.DomainMap().NumGlobalElements()),
97 ExcMessage("Number of columns in C must match dimension of A"));
98
99
100 // Ensure the MultiVector has only one column.
101 Assert((multi_vector.NumVectors() == 1),
102 ExcMessage("The MultiVector must have exactly one column."));
103
104
105 // Create diagonal matrix from first vector of v
106 // Explicitly cast the map to Epetra_Map
107 const Epetra_Map &map =
108 static_cast<const Epetra_Map &>(multi_vector.Map());
109
110
111 // Create a diagonal matrix with 1 nonzero entry per row
112 Epetra_CrsMatrix diag_matrix(Copy, map, 1);
113 for (int i = 0; i < multi_vector.Map().NumMyElements(); ++i)
114 {
115 int global_row = multi_vector.Map().GID(i);
116 double val = multi_vector[0][i]; // Access first vector
117 diag_matrix.InsertGlobalValues(global_row, 1, &val, &global_row);
118 }
119 diag_matrix.FillComplete();
120
121
122 Epetra_CrsMatrix *W =
123 new Epetra_CrsMatrix(Copy, Ct_trilinos.RowMap(), 0);
124 EpetraExt::MatrixMatrix::Multiply(
125 Ct_trilinos, false, diag_matrix, false, *W);
126
127
128 // Compute Ct^T * W, which is equivalent to (C^T * diag(V)) * C
129 Epetra_CrsMatrix *CtT_W = new Epetra_CrsMatrix(Copy, W->RangeMap(), 0);
130 EpetraExt::MatrixMatrix::Multiply(
131 *W, false /* transpose */, Ct_trilinos, true, *CtT_W);
132
133
134 // Add A to the result, scaling with gamma
135 Epetra_CrsMatrix *result =
136 new Epetra_CrsMatrix(Copy,
137 A_trilinos.RowMap(),
138 A_trilinos.MaxNumEntries());
139 EpetraExt::MatrixMatrix::Add(
140 A_trilinos, false, 1.0, *CtT_W, false, gamma, result);
141 result->FillComplete();
142
143
144 // Initialize the final Trilinos matrix
145 augmented_matrix.reinit(*result, true /*copy_values*/);
146
147
148 // Delete unnecessary objects.
149 delete W;
150 delete CtT_W;
151 delete result;
152 }
153 else
154 {

156 AssertThrow(false, ExcNotImplemented("Matrix type not supported!"));
