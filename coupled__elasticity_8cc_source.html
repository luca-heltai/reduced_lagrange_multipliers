29template <int dim, int spacedim>
30void
31CoupledElasticityProblem<dim, spacedim>::reassemble_coupling_rhs()
32{
33 TimerOutput::Scope t(this->computing_timer, "updating coupling rhs");
34
35 this->system_rhs.block(1) = 0;
36
37 std::vector<types::global_dof_index> fe_dof_indices(
38 this->fe->n_dofs_per_cell());
39 std::vector<types::global_dof_index> inclusion_dof_indices(
40 this->inclusions.n_dofs_per_inclusion());
41
42 Vector<double> local_rhs(this->inclusions.n_dofs_per_inclusion());
43
44 auto particle = this->inclusions.inclusions_as_particles.begin();
45 while (particle != this->inclusions.inclusions_as_particles.end())
46 {
47 const auto &cell = particle->get_surrounding_cell();
48 const auto dh_cell =
49 typename DoFHandler<spacedim>::cell_iterator(*cell, &this->dh);
50 dh_cell->get_dof_indices(fe_dof_indices);
51 const auto pic =
52 this->inclusions.inclusions_as_particles.particles_in_cell(cell);
53 Assert(pic.begin() == particle, ExcInternalError());
54
55 auto p = pic.begin();
56 auto next_p = pic.begin();
57 while (p != pic.end())
58 {
59 const auto inclusion_id =
60 this->inclusions.get_inclusion_id(p->get_id());
61 inclusion_dof_indices = this->inclusions.get_dof_indices(p->get_id());
62 local_rhs = 0;
63
64 // Extract all points that refer to the same inclusion
65 std::vector<Point<spacedim>> ref_q_points;
66 for (; next_p != pic.end() && this->inclusions.get_inclusion_id(
67 next_p->get_id()) == inclusion_id;
68 ++next_p)
69 ref_q_points.push_back(next_p->get_reference_location());
70 FEValues<spacedim, spacedim> fev(*this->fe,
71 ref_q_points,
72 update_values | update_gradients);
73 fev.reinit(dh_cell);
74 // double temp = 0;
75 for (unsigned int q = 0; q < ref_q_points.size(); ++q)
76 {
77 const auto id = p->get_id();
78 const auto &inclusion_fe_values =
79 this->inclusions.get_fe_values(id);
80 const auto &real_q = p->get_location();
81 const auto ds = this->inclusions.get_JxW(
82 id); // /inclusions.get_radius(inclusion_id);
83
84 // Coupling and inclusions matrix
85 for (unsigned int j = 0;
86 j < this->inclusions.n_dofs_per_inclusion();
87 ++j)
88 {
89 if (this->inclusions.data_file != "")
90 {
91 if (this->inclusions.inclusions_data[inclusion_id]
92 .size() > j)
93 {
94 auto temp =
95 inclusion_fe_values[j] * inclusion_fe_values[j] *
96 this->inclusions.get_inclusion_data(inclusion_id,
97 j) /
98 // inclusions.inclusions_data[inclusion_id][j] / //
99 // data is always prescribed in relative coordinates
100 this->inclusions.get_radius(inclusion_id) * ds;
101 if (this->par.initial_time != this->par.final_time)
102 temp *= this->inclusions.inclusions_rhs.value(
103 real_q, this->inclusions.get_component(j));
104 local_rhs(j) += temp;
105 }
106 }
107 else
108 {
109 local_rhs(j) +=
110 inclusion_fe_values[j] *
111 this->inclusions.inclusions_rhs.value(
112 real_q, this->inclusions.get_component(j)) /
113 this->inclusions.get_radius(inclusion_id) * ds;
114 }
115 }
116 ++p;
117 }
118 // I expect p and next_p to be the same now.
119 Assert(p == next_p, ExcInternalError());
120 // Add local matrices to global ones
121 this->inclusion_constraints.distribute_local_to_global(
122 local_rhs, inclusion_dof_indices, this->system_rhs.block(1));
123 }
124 particle = pic.end();
125 }
126 this->system_rhs.compress(VectorOperation::add);
127}

148template <int dim, int spacedim>
149void
150CoupledElasticityProblem<dim, spacedim>::run_timestep()
151{
152 if (this->cycle == 0) // at first timestep we refine
153 {
154 if (this->par.refinement_strategy == "inclusions")
155 {
156 this->refine_and_transfer_around_inclusions();
157 std::cout << "refining around inclusions" << std::endl;
158
159 this->assemble_elasticity_system(); // we need to refine A too
160 this->assemble_coupling();
161 this->solve();
162 }
163 else
164 {
165 for (unsigned int ref_cycle = 0;
166 ref_cycle < this->par.n_refinement_cycles;
167 ++ref_cycle)
168 {
169 this->assemble_elasticity_system(); // questo mi serve perche sto
170 // raffinando
171 this->assemble_coupling();
172 this->solve();
173 if (ref_cycle != this->par.n_refinement_cycles - 1)
174 this->refine_and_transfer();
175 }
176 }
177 }
178 else
179 {
180 reassemble_coupling_rhs();
181 this->solve();
182 }
183
184
185 // if (this->par.output_results)
186 // this->output_results();
187
188 coupling_pressure.clear();
189 coupling_pressure_at_inclusions.clear();
190 // coupling_pressure = output_pressure(cycle == 0 ? true : false);
191 // compute_coupling_pressure();
192 // output_coupling_pressure(cycle == 0 ? true : false);
193
194 this->compute_internal_and_boundary_stress(this->cycle == 0 ? true : false);
195 this->cycle++;
196}

206template <int dim, int spacedim>
207void
208CoupledElasticityProblem<dim, spacedim>::update_inclusions_data(
209 std::vector<double> new_data,
210 std::vector<int> cells_per_vessel)
211{
212 Assert(cells_per_vessel.size() == this->inclusions.get_n_vessels(),
213 ExcInternalError());
214 std::vector<std::vector<double>> full_vector;
215 unsigned int starting_point = 0;
216 for (auto &N1 : cells_per_vessel)
217 {
218 std::vector<double> vessel_vector;
219 for (auto j = 0; j < N1; ++j)
220 {
221 AssertIndexRange(starting_point + j, new_data.size());
222 vessel_vector.push_back(new_data[starting_point + j]);
223 }
224 starting_point += N1;
225
226 full_vector.push_back(vessel_vector);
227 }
228 this->inclusions.update_inclusions_data(full_vector);
229}

231template <int dim, int spacedim>
232std::vector<std::vector<double>>
233CoupledElasticityProblem<dim, spacedim>::split_pressure_over_inclusions(
234 std::vector<int> number_of_cells_per_vessel,
235 Vector<double> /* full_press */) const
236{
237 Assert(number_of_cells_per_vessel.size() == this->inclusions.get_n_vessels(),
238 ExcInternalError());
239
240 std::vector<std::vector<double>> split_pressure;
241 unsigned starting_inclusion = 0;
242
243 for (unsigned int vessel = 0; vessel < number_of_cells_per_vessel.size();
244 ++vessel)
245 {
246 auto N2 = number_of_cells_per_vessel[vessel];
247 auto N1 = this->inclusions.get_inclusions_in_vessel(vessel);
248
249 std::vector<double> new_vector;
250
251 // const std::vector<double> pressure_of_inc_in_vessel(pressure);//
252 // (*pressure[starting_inclusion], *pressure[starting_inclusion+N1]);
253 const Vector<double> pressure_of_inc_in_vessel(
254 coupling_pressure_at_inclusions);
255
256 new_vector.push_back(pressure_of_inc_in_vessel[starting_inclusion]);
257
258 for (auto cell = 1; cell < N2 - 1; ++cell)
259 {
260 auto X = cell / (N2 - 1) * (N1 - 1);
261 auto j = floor(X);
262 auto w = X - j;
263
264 new_vector.push_back(
265 (1 - w) * pressure_of_inc_in_vessel[starting_inclusion + j] +
266 (w)*pressure_of_inc_in_vessel[starting_inclusion + j + 1]);
267 }
268 new_vector.push_back(
269 pressure_of_inc_in_vessel[starting_inclusion + N1 - 1]);
270 starting_inclusion += N1;
271 split_pressure.push_back(new_vector);
272 }
273
274 return split_pressure;
275}

278template <int dim, int spacedim>
279void
280CoupledElasticityProblem<dim, spacedim>::refine_and_transfer_around_inclusions()
281{
282 TimerOutput::Scope t(this->computing_timer, "Refine");
283 Vector<float> error_per_cell(this->tria.n_active_cells());
284 KellyErrorEstimator<spacedim>::estimate(
285 this->dh,
286 QGauss<spacedim - 1>(this->par.fe_degree + 1),
287 {},
288 this->locally_relevant_solution.block(0),
289 error_per_cell);
290
291 const int material_id = 1;
292
293 auto particle = this->inclusions.inclusions_as_particles.begin();
294 while (particle != this->inclusions.inclusions_as_particles.end())
295 {
296 const auto &cell = particle->get_surrounding_cell();
297 const auto pic =
298 this->inclusions.inclusions_as_particles.particles_in_cell(cell);
299 Assert(pic.begin() == particle, ExcInternalError());
300
301 cell->set_refine_flag();
302 cell->set_material_id(material_id);
303
304 // for (auto f : cell->face_indices())
305 // for (unsigned int face_no = 0; face_no
306 // <GeometryInfo<spacedim>::faces_per_cell; ++face_no)
307 for (auto vertex : cell->vertex_indices())
308 {
309 for (const auto &neigh_i :
310 GridTools::find_cells_adjacent_to_vertex(this->dh, vertex))
311 // for (auto neigh_i = cell->neighbor(face_no)->begin_face();
312 // neigh_i < cell->neighbor(face_no)->end_face(); ++neigh_i)
313 {
314 if (!neigh_i->refine_flag_set())
315 {
316 neigh_i->set_refine_flag();
317 neigh_i->set_material_id(material_id);
318 for (auto vertey : neigh_i->vertex_indices())
319 for (const auto &neigh_j :
320 GridTools::find_cells_adjacent_to_vertex(this->dh,
321 vertey))
322 // for (auto neigh_j =
323 // cell->neighbor(neigh_i)->begin_face(); neigh_j <
324 // cell->neighbor(neigh_i)->end_face(); ++neigh_j)
325 {
326 neigh_j->set_refine_flag();
327 neigh_j->set_material_id(material_id);
328 }
329 }
330 }
331 }
332
333 particle = pic.end();
334 }
335 this->execute_actual_refine_and_transfer();
336
337 for (unsigned int ref_cycle = 0;
338 ref_cycle < this->par.n_refinement_cycles - 1;
339 ++ref_cycle)
340 {
341 for (const auto &cell : this->tria.active_cell_iterators())
342 {
343 if (cell->material_id() == material_id)
344 cell->set_refine_flag();
345 }
346 this->execute_actual_refine_and_transfer();
347 }
348}
