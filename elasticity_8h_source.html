125template <int dim>
126class RigidBodyMotion : public Function<dim>
127{
128public:
129 RigidBodyMotion(const unsigned int type_);
130
131 virtual double
132 value(const Point<dim> &p, const unsigned int component) const override;
133
134private:
135 const unsigned int type;
136};
137
138
139template <int dim>
140RigidBodyMotion<dim>::RigidBodyMotion(const unsigned int _type)
141 : Function<dim>(dim)
142 , type(_type)
143{
144 Assert(dim == 2 || dim == 3, ExcNotImplemented());
145 Assert((dim == 2 && type <= 2) || (dim == 3 && type <= 5),
146 ExcNotImplemented());
147}
148
149
150
151template <int dim>
152double
153RigidBodyMotion<dim>::value(const Point<dim> &p,
154 const unsigned int component) const
155{
156 if constexpr (dim == 2)
157 {
158 // 2D rigid body modes: 2 translations and 1 rotation
159 const std::array<double, 3> modes{{static_cast<double>(component == 0),
160 static_cast<double>(component == 1),
161 (component == 0) ? -p[1] :
162 (component == 1) ? p[0] :
163 0.}};
164
165 return modes[type];
166 }
167 else // dim == 3
168 {
169 // 3D rigid body modes: 3 translations and 3 rotations
170 const std::array<double, 6> modes{{static_cast<double>(component == 0),
171 static_cast<double>(component == 1),
172 static_cast<double>(component == 2),
173 (component == 0) ? 0. :
174 (component == 1) ? p[2] :
175 -p[1],
176 (component == 0) ? -p[2] :
177 (component == 1) ? 0. :
178 p[0],
179 (component == 0) ? p[1] :
180 (component == 1) ? -p[0] :
181 0.}};
182
183 return modes[type];
184 }
185}

248template <int dim, int spacedim>
249ElasticityProblemParameters<dim, spacedim>::ElasticityProblemParameters()
250 : ParameterAcceptor("/Immersed Problem/")
251 , rhs("/Immersed Problem/Right hand side", spacedim)
252 , exact_solution("/Immersed Problem/Exact solution", spacedim)
253 , bc("/Immersed Problem/Dirichlet boundary conditions", spacedim)
254 , Neumann_bc("/Immersed Problem/Neumann boundary conditions", spacedim)
255 , inner_control("/Immersed Problem/Solver/Inner control")
256 , outer_control("/Immersed Problem/Solver/Outer control")
257 , convergence_table(std::vector<std::string>(spacedim, "u"))
258{
259 add_parameter("FE degree", fe_degree, "", this->prm, Patterns::Integer(1));
260 add_parameter("Output directory", output_directory);
261 add_parameter("Output name", output_name);
262 add_parameter("Output results also before solving",
263 output_results_before_solving);
264 add_parameter("Initial refinement", initial_refinement);
265 add_parameter("Dirichlet boundary ids", dirichlet_ids);
266 add_parameter("Weak Dirichlet boundary ids", weak_dirichlet_ids);
267 add_parameter("Neumann boundary ids", neumann_ids);
268 add_parameter("Normal flux boundary ids", normal_flux_ids);
269 add_parameter("Output pressure", output_pressure);
270 enter_subsection("Grid generation");
271 {
272 add_parameter("Domain type",
273 domain_type,
274 "",
275 this->prm,
276 Patterns::Selection("generate|file|cheese|cylinder"));
277 add_parameter("Grid generator", name_of_grid);
278 add_parameter("Grid generator arguments", arguments_for_grid);
279 }
280 leave_subsection();
281 enter_subsection("Refinement and remeshing");
282 {
283 add_parameter("Strategy",
284 refinement_strategy,
285 "",
286 this->prm,
287 Patterns::Selection("fixed_fraction|fixed_number|global"));
288 add_parameter("Coarsening fraction", coarsening_fraction);
289 add_parameter("Refinement fraction", refinement_fraction);
290 add_parameter("Maximum number of cells", max_cells);
291 add_parameter("Number of refinement cycles", n_refinement_cycles);
292 }
293 leave_subsection();
294 enter_subsection("Physical constants");
295 {
296 add_parameter("Lame mu", Lame_mu);
297 add_parameter("Lame lambda", Lame_lambda);
298 }
299 leave_subsection();
300 enter_subsection("Material properties");
301 add_parameter("Material tags by material id", material_tags_by_material_id);
302 leave_subsection();
303 enter_subsection("Exact solution");
304 {
305 add_parameter("Weight expression", weight_expression);
306 }
307 leave_subsection();
308 enter_subsection("Time dependency");
309 {
310 add_parameter("Initial time", initial_time);
311 add_parameter("Final time", final_time);
312 add_parameter("Time step", dt);
313 }
314 leave_subsection();
315
316 this->prm.enter_subsection("Error");
317 convergence_table.add_parameters(this->prm);
318 this->prm.leave_subsection();
319
320 auto reset_function = [this]() {
321 this->prm.set("Function expression", (spacedim == 2 ? "0; 0" : "0; 0; 0"));
322 };
323 rhs.declare_parameters_call_back.connect(reset_function);
324 exact_solution.declare_parameters_call_back.connect(reset_function);
325 Neumann_bc.declare_parameters_call_back.connect(reset_function);
326 bc.declare_parameters_call_back.connect(reset_function);
327
328 this->parse_parameters_call_back.connect([this]() {
329 if (!material_properties_by_id.empty() ||
330 material_tags_by_material_id.empty())
331 {
332 // Already initialized. Just check consistency.
333 AssertDimension(material_properties_by_id.size(),
334 material_tags_by_material_id.size());
335 return;
336 }
337 // Make sure we exit our subsection first, so that registration of the new
338 // classes works properly.
339 this->leave_my_subsection(this->prm);
340 for (const auto &[material_id, tag] : material_tags_by_material_id)
341 {
342 material_properties_by_id[material_id] =
343 std::make_unique<MaterialProperties>(tag);
344 }
345 // Go back to our subsection, so we can continue parsing parameters.
346 this->enter_my_subsection(this->prm);
347 });
348}
