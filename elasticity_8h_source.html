169template <int dim, int spacedim>
170ElasticityProblemParameters<dim, spacedim>::ElasticityProblemParameters()
171 : ParameterAcceptor("/Immersed Problem/")
172 , rhs("/Immersed Problem/Right hand side", spacedim)
173 , exact_solution("/Immersed Problem/Exact solution", spacedim)
174 , bc("/Immersed Problem/Dirichlet boundary conditions", spacedim)
175 , Neumann_bc("/Immersed Problem/Neumann boundary conditions", spacedim)
176 , inner_control("/Immersed Problem/Solver/Inner control")
177 , outer_control("/Immersed Problem/Solver/Outer control")
178 , convergence_table(std::vector<std::string>(spacedim, "u"))
179{
180 add_parameter("FE degree", fe_degree, "", this->prm, Patterns::Integer(1));
181 add_parameter("Output directory", output_directory);
182 add_parameter("Output name", output_name);
183 add_parameter("Output results also before solving",
184 output_results_before_solving);
185 add_parameter("Initial refinement", initial_refinement);
186 add_parameter("Dirichlet boundary ids", dirichlet_ids);
187 add_parameter("Neumann boundary ids", neumann_ids);
188 add_parameter("Normal flux boundary ids", normal_flux_ids);
189 add_parameter("Output pressure", output_pressure);
190 enter_subsection("Grid generation");
191 {
192 add_parameter("Domain type",
193 domain_type,
194 "",
195 this->prm,
196 Patterns::Selection("generate|file|cheese|cylinder"));
197 add_parameter("Grid generator", name_of_grid);
198 add_parameter("Grid generator arguments", arguments_for_grid);
199 }
200 leave_subsection();
201 enter_subsection("Refinement and remeshing");
202 {
203 add_parameter("Strategy",
204 refinement_strategy,
205 "",
206 this->prm,
207 Patterns::Selection("fixed_fraction|fixed_number|global"));
208 add_parameter("Coarsening fraction", coarsening_fraction);
209 add_parameter("Refinement fraction", refinement_fraction);
210 add_parameter("Maximum number of cells", max_cells);
211 add_parameter("Number of refinement cycles", n_refinement_cycles);
212 }
213 leave_subsection();
214 enter_subsection("Physical constants");
215 {
216 add_parameter("Lame mu", Lame_mu);
217 add_parameter("Lame lambda", Lame_lambda);
218 }
219 leave_subsection();
220 enter_subsection("Exact solution");
221 {
222 add_parameter("Weight expression", weight_expression);
223 }
224 leave_subsection();
225 enter_subsection("Time dependency");
226 {
227 add_parameter("Initial time", initial_time);
228 add_parameter("Final time", final_time);
229 add_parameter("Time step", dt);
230 }
231 leave_subsection();
232
233 this->prm.enter_subsection("Error");
234 convergence_table.add_parameters(this->prm);
235 this->prm.leave_subsection();
236
237 auto reset_function = [this]() {
238 this->prm.set("Function expression", (spacedim == 2 ? "0; 0" : "0; 0; 0"));
239 };
240 rhs.declare_parameters_call_back.connect(reset_function);
241 exact_solution.declare_parameters_call_back.connect(reset_function);
242 Neumann_bc.declare_parameters_call_back.connect(reset_function);
243 bc.declare_parameters_call_back.connect(reset_function);
244}
