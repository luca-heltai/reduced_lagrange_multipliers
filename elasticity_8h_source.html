126 unsigned int fe_degree = 1;
127 unsigned int initial_refinement = 5;

129 std::list<types::boundary_id> neumann_ids{};
130 std::set<types::boundary_id> normal_flux_ids{};
131 std::string domain_type = "generate";
132 std::string name_of_grid = "hyper_cube";
133 std::string arguments_for_grid = "-1: 1: false";
134 std::string refinement_strategy = "fixed_fraction";
135 double coarsening_fraction = 0.0;
136 double refinement_fraction = 0.3;

139 bool output_pressure = false;
140
141 double Lame_mu = 1;
142 double Lame_lambda = 1;

144 mutable ParameterAcceptorProxy<Functions::ParsedFunction<spacedim>> rhs;
145 mutable ParameterAcceptorProxy<Functions::ParsedFunction<spacedim>>
146 exact_solution;
147 mutable ParameterAcceptorProxy<Functions::ParsedFunction<spacedim>> bc;
148 mutable ParameterAcceptorProxy<Functions::ParsedFunction<spacedim>>

151 std::string weight_expression = "1.";

156 bool output_results_before_solving = false;
157

159

168template <int dim, int spacedim>
169ElasticityProblemParameters<dim, spacedim>::ElasticityProblemParameters()
170 : ParameterAcceptor("/Immersed Problem/")
171 , rhs("/Immersed Problem/Right hand side", spacedim)
172 , exact_solution("/Immersed Problem/Exact solution", spacedim)
173 , bc("/Immersed Problem/Dirichlet boundary conditions", spacedim)
174 , Neumann_bc("/Immersed Problem/Neumann boundary conditions", spacedim)
175 , inner_control("/Immersed Problem/Solver/Inner control")
176 , outer_control("/Immersed Problem/Solver/Outer control")
177 , convergence_table(std::vector<std::string>(spacedim, "u"))
178{
179 add_parameter("FE degree", fe_degree, "", this->prm, Patterns::Integer(1));
180 add_parameter("Output directory", output_directory);
181 add_parameter("Output name", output_name);
182 add_parameter("Output results also before solving",
183 output_results_before_solving);
184 add_parameter("Initial refinement", initial_refinement);
185 add_parameter("Dirichlet boundary ids", dirichlet_ids);
186 add_parameter("Neumann boundary ids", neumann_ids);
187 add_parameter("Normal flux boundary ids", normal_flux_ids);
188 add_parameter("Output pressure", output_pressure);
189 enter_subsection("Grid generation");
190 {
191 add_parameter("Domain type",
192 domain_type,
193 "",
194 this->prm,
195 Patterns::Selection("generate|file|cheese|cylinder"));
196 add_parameter("Grid generator", name_of_grid);
197 add_parameter("Grid generator arguments", arguments_for_grid);
198 }
199 leave_subsection();
200 enter_subsection("Refinement and remeshing");
201 {
202 add_parameter("Strategy",
203 refinement_strategy,
204 "",
205 this->prm,
206 Patterns::Selection("fixed_fraction|fixed_number|global"));
207 add_parameter("Coarsening fraction", coarsening_fraction);
208 add_parameter("Refinement fraction", refinement_fraction);
209 add_parameter("Maximum number of cells", max_cells);
210 add_parameter("Number of refinement cycles", n_refinement_cycles);
211 }
212 leave_subsection();
213 enter_subsection("Physical constants");
214 {
215 add_parameter("Lame mu", Lame_mu);
216 add_parameter("Lame lambda", Lame_lambda);
217 }
218 leave_subsection();
219 enter_subsection("Exact solution");
220 {
221 add_parameter("Weight expression", weight_expression);
222 }
223 leave_subsection();
224 enter_subsection("Time dependency");
225 {
226 add_parameter("Initial time", initial_time);
227 add_parameter("Final time", final_time);
228 add_parameter("Time step", dt);
229 }
230 leave_subsection();
231
232 this->prm.enter_subsection("Error");
233 convergence_table.add_parameters(this->prm);
234 this->prm.leave_subsection();
235
236 auto reset_function = [this]() {
237 this->prm.set("Function expression", (spacedim == 2 ? "0; 0" : "0; 0; 0"));
238 };
239 rhs.declare_parameters_call_back.connect(reset_function);
240 exact_solution.declare_parameters_call_back.connect(reset_function);
241 Neumann_bc.declare_parameters_call_back.connect(reset_function);
242 bc.declare_parameters_call_back.connect(reset_function);
243}
