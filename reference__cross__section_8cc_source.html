57template <int dim, int spacedim, int n_components>
58void
59ReferenceCrossSection<dim, spacedim, n_components>::make_grid()
60
61{
62 if (par.inclusion_type == "hyper_ball")
63 {
64 if constexpr (dim == 1 && spacedim == 3)
65 {
66 // 1D inclusion in 3D space. A disk.
67 Triangulation<1, 2> tria_12;
68 GridGenerator::hyper_sphere(tria_12);
69 GridGenerator::flatten_triangulation(tria_12, triangulation);
70 triangulation.set_all_manifold_ids(1);
71 triangulation.set_manifold(1, PolarManifold<dim, spacedim>());
72 }
73 else if constexpr (dim == spacedim - 1)
74 GridGenerator::hyper_sphere(triangulation);
75 else
76 GridGenerator::hyper_ball(triangulation);
77 }
78 else if (par.inclusion_type == "hyper_cube")
79 GridGenerator::hyper_cube(triangulation, -1, 1);
80 else
81 {
82 AssertThrow(false,
83 ExcMessage("Unknown inclusion type. Please check the "
84 "parameter file."));
85 }
86 triangulation.refine_global(par.refinement_level);
87}

90template <int dim, int spacedim, int n_components>
91void
92ReferenceCrossSection<dim, spacedim, n_components>::setup_dofs()
93{
94 dof_handler.distribute_dofs(fe);
95 DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
96 DoFTools::make_sparsity_pattern(dof_handler, dsp);
97 sparsity_pattern.copy_from(dsp);
98 sparsity_pattern.compress();
99 mass_matrix.reinit(sparsity_pattern);
100 MatrixTools::create_mass_matrix(dof_handler, quadrature_formula, mass_matrix);
101
102 const auto n_global_q_points =
103 triangulation.n_active_cells() * quadrature_formula.size();
104
105 std::vector<Point<spacedim>> points;
106 std::vector<double> weights;
107 points.reserve(n_global_q_points);
108 weights.reserve(n_global_q_points);
109
110 FEValues<dim, spacedim> fe_values(mapping,
111 fe,
112 quadrature_formula,
113 update_quadrature_points |
114 update_JxW_values);
115 for (const auto &cell : dof_handler.active_cell_iterators())
116 {
117 fe_values.reinit(cell);
118 points.insert(points.end(),
119 fe_values.get_quadrature_points().begin(),
120 fe_values.get_quadrature_points().end());
121 weights.insert(weights.end(),
122 fe_values.get_JxW_values().begin(),
123 fe_values.get_JxW_values().end());
124 }
125 global_quadrature = Quadrature<spacedim>(points, weights);
126}

129template <int dim, int spacedim, int n_components>
130void
131ReferenceCrossSection<dim, spacedim, n_components>::compute_basis()
132{
133 basis_functions.resize(polynomials.n() * n_components,
134 Vector<double>(dof_handler.n_dofs()));
135 for (unsigned int i = 0; i < basis_functions.size(); ++i)
136 {
137 const auto comp_i = i % n_components;
138 const auto poly_i = i / n_components;
139 VectorFunctionFromScalarFunctionObject<spacedim> function(
140 [&](const Point<spacedim> &p) {
141 return polynomials.compute_value(poly_i, p);
142 },
143 comp_i,
144 n_components);
145 VectorTools::interpolate(dof_handler, function, basis_functions[i]);
146 }
147 // Grahm-Schmidt orthogonalization
148 std::vector<bool> is_zero(basis_functions.size(), false);
149 for (unsigned int i = 0; i < basis_functions.size(); ++i)
150 {
151 for (unsigned int j = 0; j < i; ++j)
152 {
153 const auto coeff =
154 mass_matrix.matrix_scalar_product(basis_functions[i],
155 basis_functions[j]);
156 basis_functions[i].sadd(1, -coeff, basis_functions[j]);
157 }
158 const auto coeff = mass_matrix.matrix_scalar_product(basis_functions[i],
159 basis_functions[i]);
160
161 // If the basis functions are on a plane, and we are in 3d, they may be
162 // zero
163 if (coeff > 1e-10)
164 basis_functions[i] /= std::sqrt(coeff);
165 else
166 is_zero[i] = true;
167 }
168
169 // Remove the zero basis functions
170 for (int i = is_zero.size() - 1; i >= 0; --i)
171 if (is_zero[i] == true)
172 basis_functions.erase(basis_functions.begin() + i);
173
174 // functions
175 if (par.selected_coefficients.empty())
176 {
177 par.selected_coefficients.resize(basis_functions.size());
178 std::iota(par.selected_coefficients.begin(),
179 par.selected_coefficients.end(),
180 0);
181 }
182 else
183 {
184 for (const auto &index : par.selected_coefficients)
185 {
186 AssertThrow(index < basis_functions.size(),
187 ExcIndexRange(index, 0, basis_functions.size()));
188 }
189 }
190
191 selected_basis_functions.resize(par.selected_coefficients.size(),
192 Vector<double>(dof_handler.n_dofs()));
193 // Compute the selected basis functions as the rows of the inverse metric
194 // times the basis functions
195 for (unsigned int i = 0; i < par.selected_coefficients.size(); ++i)
196 {
197 selected_basis_functions[i] =
198 basis_functions[par.selected_coefficients[i]];
199 }
200
201
202 // Now compute the Matrix of the selected basis functions evaluated on the
203 // quadrature points
204 basis_functions_on_qpoints.reinit(global_quadrature.size(),
205 selected_basis_functions.size() *
206 n_components);
207
208 std::vector<Vector<double>> values(quadrature_formula.size(),
209 Vector<double>(fe.n_components()));
210
211 FEValues<dim, spacedim> fe_values(mapping,
212 fe,
213 quadrature_formula,
214 update_values);
215
216 for (const auto &cell : dof_handler.active_cell_iterators())
217 {
218 fe_values.reinit(cell);
219 const unsigned int shift =
220 cell->global_active_cell_index() * quadrature_formula.size();
221
222 for (unsigned int i = 0; i < selected_basis_functions.size(); ++i)
223 {
224 fe_values.get_function_values(selected_basis_functions[i], values);
225 for (unsigned int q = 0; q < quadrature_formula.size(); ++q)
226 {
227 for (unsigned int comp = 0; comp < n_components; ++comp)
228 {
229 basis_functions_on_qpoints(q + shift,
230 i * n_components + comp) =
231 values[q][comp];
232 }
233 }
234 }
235 }
236}

258template <int dim, int spacedim, int n_components>
259const double &
260ReferenceCrossSection<dim, spacedim, n_components>::shape_value(
261 const unsigned int i,
262 const unsigned int q,
263 const unsigned int comp) const
264{
265 AssertIndexRange(i, selected_basis_functions.size());
266 AssertIndexRange(q, global_quadrature.size());
267 AssertIndexRange(comp, n_components);
268
269 // Check that the matrix is not empty
270 AssertThrow(!basis_functions_on_qpoints.empty(),
271 ExcMessage(
272 "The basis functions on quadrature points are empty."));
273 return basis_functions_on_qpoints(q, i * n_components + comp);
274};

325template <int dim, int spacedim, int n_components>
326Quadrature<spacedim>
327ReferenceCrossSection<dim, spacedim, n_components>::get_transformed_quadrature(
328 const Point<spacedim> &new_origin,
329 const Tensor<1, spacedim> &new_vertical,
330 const double scale) const
331{
332 AssertThrow(new_vertical.norm() > 0,
333 ExcMessage("The new vertical direction must be non-zero."));
334
335 Tensor<2, spacedim> rotation;
336 Tensor<1, spacedim> vertical;
337 vertical[spacedim - 1] = 1;
338 if constexpr (spacedim == 3)
339 {
340 Tensor<1, spacedim> axis = cross_product_3d(vertical, new_vertical);
341 const double axis_norm = axis.norm();
342 if (axis_norm < 1e-10)
343 {
344 // The two vectors are parallel, no rotation needed
345 for (unsigned int i = 0; i < spacedim; ++i)
346 rotation[i][i] = 1;
347 }
348 else
349 {
350 axis /= axis_norm;
351 double angle = Physics::VectorRelations::signed_angle(vertical,
352 new_vertical,
353 axis);
354 rotation =
355 Physics::Transformations::Rotations::rotation_matrix_3d(axis,
356 angle);
357 }
358 }
359 else if constexpr (spacedim == 2)
360 {
361 double angle = Physics::VectorRelations::angle(vertical, new_vertical);
362 rotation = Physics::Transformations::Rotations::rotation_matrix_2d(angle);
363 }
364
365 // Create transformed quadrature by applying the mapping to points and weights
366 std::vector<Point<spacedim>> transformed_points(global_quadrature.size());
367 std::vector<double> transformed_weights(global_quadrature.size());
368
369 const auto &original_points = global_quadrature.get_points();
370 const auto &original_weights = global_quadrature.get_weights();
371
372 // Calculate scale factor for weights
373 const double weight_scale_factor = std::pow(scale, dim);
374
375 for (unsigned int q = 0; q < original_points.size(); ++q)
376 {
377 transformed_points[q] =
378 rotation * (scale * original_points[q]) + new_origin;
379
380 // Scale the weight by scale^dim
381 transformed_weights[q] = original_weights[q] * weight_scale_factor;
382 }
383
384 return Quadrature<spacedim>(transformed_points, transformed_weights);
385}
