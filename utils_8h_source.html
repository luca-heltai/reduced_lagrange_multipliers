52struct RefinementParameters : public ParameterAcceptor
53{
54 RefinementParameters()
55 : ParameterAcceptor("Local refinement parameters")
56 {
57 this->add_parameter("Refinement strategy",
58 refinement_strategy,
59 "",
60 this->prm,
61 Patterns::Selection("space|embedded"));
62 this->add_parameter("Space post-refinement cycles",
63 space_post_refinement_cycles);
64 this->add_parameter("Embedded post-refinement cycles",
65 embedded_post_refinement_cycles);
66 this->add_parameter("Space pre-refinement cycles",
67 space_pre_refinement_cycles);
68 this->add_parameter("Embedded pre-refinement cycles",
69 embedded_pre_refinement_cycles);
70 this->add_parameter("Refinement factor", refinement_factor);
71 this->add_parameter("Max refinement level", max_refinement_level);
72 }
73
74 std::string refinement_strategy = "space";
75 unsigned int space_post_refinement_cycles = 0;
76 unsigned int embedded_post_refinement_cycles = 0;
77 unsigned int space_pre_refinement_cycles = 0;
78 unsigned int embedded_pre_refinement_cycles = 0;
79 double refinement_factor = 1.0;
80 int max_refinement_level = 10;
81};

83template <int dim, int spacedim>
84inline void
85read_grid_and_cad_files(const std::string &grid_file_name,
86 const std::string &ids_and_cad_file_names,
87 Triangulation<dim, spacedim> &tria)
88{
89 GridIn<dim, spacedim> grid_in;
90 grid_in.attach_triangulation(tria);
91 grid_in.read(grid_file_name);
92
93#ifdef DEAL_II_WITH_OPENCASCADE
94 using map_type = std::map<types::manifold_id, std::string>;
95 using Converter = Patterns::Tools::Convert<map_type>;
96 for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
97 {
98 const auto &manifold_id = pair.first;
99 const auto &cad_file_name = pair.second;
100 const auto extension = boost::to_lower_copy(
101 cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
102 TopoDS_Shape shape;
103 if (extension == "iges" || extension == "igs")
104 shape = OpenCASCADE::read_IGES(cad_file_name);
105 else if (extension == "step" || extension == "stp")
106 shape = OpenCASCADE::read_STEP(cad_file_name);
107 else
108 AssertThrow(false,
109 ExcNotImplemented("We found an extension that we "
110 "do not recognize as a CAD file "
111 "extension. Bailing out."));
112 const auto n_elements = OpenCASCADE::count_elements(shape);
113 if ((std::get<0>(n_elements) == 0))
114 tria.set_manifold(
115 manifold_id,
116 OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
117 else if (spacedim == 3)
118 {
119 const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
120 t->set_manifold(manifold_id,
121 OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
122 shape));
123 }
124 else
125 tria.set_manifold(manifold_id,
126 OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
127 TopoDS::Face(shape)));
128 }
129#else
130 (void)ids_and_cad_file_names;
131#endif
132}

136template <int reduced_dim, int spacedim>
137void
138adjust_grids(Triangulation<spacedim, spacedim> &space_triangulation,
139 Triangulation<reduced_dim, spacedim> &embedded_triangulation,
140 const RefinementParameters &parameters = RefinementParameters())
141{
142 Assert(
143 (dynamic_cast<parallel::TriangulationBase<reduced_dim, spacedim> *>(
144 &embedded_triangulation) == nullptr),
145 ExcMessage(
146 "The embedded triangulation must not be distributed. It will be partitioned later."));
147
148 namespace bgi = boost::geometry::index;
149
150 space_triangulation.refine_global(parameters.space_pre_refinement_cycles);
151 embedded_triangulation.refine_global(
152 parameters.embedded_pre_refinement_cycles);
153
154 // build caches so that we can get local trees
155 GridTools::Cache<spacedim, spacedim> space_cache{space_triangulation};
156 GridTools::Cache<reduced_dim, spacedim> embedded_cache{
157 embedded_triangulation};
158
159 auto refine = [&]() {
160 bool done = false;
161 bool global_done = false;
162
163 double min_embedded = 1e10;
164 double max_embedded = 0;
165 double min_space = 1e10;
166 double max_space = 0;
167
168 // bounding box
169 const bool use_space = parameters.refinement_strategy == "space";
170 const bool use_embedded = parameters.refinement_strategy == "embedded";
171
172 AssertThrow(use_space || use_embedded,
173 ExcMessage("One of the two must be true"));
174 unsigned int n_space_cells = space_triangulation.n_global_active_cells();
175 unsigned int n_embedded_cells =
176 embedded_triangulation.n_global_active_cells();
177 while (global_done == false)
178 {
179 done = true;
180 // Bounding boxes of the space grid
181 const auto &tree =
182 space_cache.get_locally_owned_cell_bounding_boxes_rtree();
183
184 const auto &embedded_tree =
185 embedded_cache.get_cell_bounding_boxes_rtree();
186
187 unsigned int n_refs = 0;
188
189 for (const auto &[embedded_box, embedded_cell] : embedded_tree)
190 {
191 const auto &[p1, p2] = embedded_box.get_boundary_points();
192 const auto diameter = p1.distance(p2);
193 min_embedded = std::min(min_embedded, diameter);
194 max_embedded = std::max(max_embedded, diameter);
195
196 for (const auto &[space_box, space_cell] :
197 tree | bgi::adaptors::queried(bgi::intersects(embedded_box)))
198 {
199 const auto &[sp1, sp2] = space_box.get_boundary_points();
200 const auto space_diameter = sp1.distance(sp2);
201 min_space = std::min(min_space, space_diameter);
202 max_space = std::max(max_space, space_diameter);
203
204 if (use_embedded &&
205 embedded_cell->level() < parameters.max_refinement_level &&
206 parameters.refinement_factor * space_diameter < diameter)
207 {
208 embedded_cell->set_refine_flag();
209 ++n_refs;
210 done = false;
211 }
212 if (use_space &&
213 space_cell->level() < parameters.max_refinement_level &&
214 parameters.refinement_factor * diameter < space_diameter)
215 {
216 space_cell->set_refine_flag();
217 ++n_refs;
218 done = false;
219 }
220 }
221 }
222 deallog << "Cells marked for refinement: " << n_refs;
223 // Synchronize done variable across all processes, otherwise we might
224 // deadlock
225 global_done =
226 Utilities::MPI::min(static_cast<int>(done),
227 space_triangulation.get_communicator());
228
229 if (global_done == false)
230 {
231 if (use_embedded)
232 {
233 n_embedded_cells =
234 embedded_triangulation.n_global_active_cells();
235 deallog << " out of " << n_embedded_cells
236 << " (embedded) cells." << std::endl;
237 embedded_triangulation.execute_coarsening_and_refinement();
238 if (n_embedded_cells ==
239 embedded_triangulation.n_global_active_cells())
240 break;
241 }
242 if (use_space)
243 {
244 n_space_cells = space_triangulation.n_global_active_cells();
245 deallog << " out of " << n_space_cells << " (space) cells."
246 << std::endl;
247 space_triangulation.execute_coarsening_and_refinement();
248 if (n_space_cells ==
249 space_triangulation.n_global_active_cells())
250 break;
251 }
252 }
253 }
254
255 deallog << std::setw(20) << std::left << "Min space: " << std::setw(12)
256 << std::right << min_space << std::setw(20) << std::left
257 << ", max space: " << std::setw(12) << std::right << max_space
258 << std::setw(25) << std::left << ", min embedded: " << std::setw(12)
259 << std::right << min_embedded << std::setw(25) << std::left
260 << ", max embedded: " << std::setw(12) << std::right << max_embedded
261 << std::endl;
262
263 return std::make_tuple(min_space, max_space, min_embedded, max_embedded);
264 };
265
266 // Do the refinement loop once, to make sure we satisfy our criterions
267 refine();
268
269
270 // Pre refine the space grid according to the delta refinement
271 if (parameters.space_post_refinement_cycles > 0)
272 for (unsigned int i = 0; i < parameters.space_post_refinement_cycles; ++i)
273 {
274 const auto &tree =
275 space_cache.get_locally_owned_cell_bounding_boxes_rtree();
276
277 const auto &embedded_tree =
278 embedded_cache.get_cell_bounding_boxes_rtree();
279
280 for (const auto &[embedded_box, embedded_cell] : embedded_tree)
281 for (const auto &[space_box, space_cell] :
282 tree | bgi::adaptors::queried(bgi::intersects(embedded_box)))
283 space_cell->set_refine_flag();
284 space_triangulation.execute_coarsening_and_refinement();
285
286 // Make sure again we satisfy our criterion after the space
287 // refinement
288 refine();
289 }
290
291 embedded_triangulation.refine_global(
292 parameters.embedded_post_refinement_cycles);
293
294 // Check once again we satisfy our criterion, and record min/max
295 const auto [sm, sM, em, eM] = refine();
296
297
298 if (Utilities::MPI::this_mpi_process(
299 space_triangulation.get_communicator()) == 0)
300 std::cout << "Space local min/max diameters : " << sm << "/" << sM
301 << std::endl
302 << "Embedded space min/max diameters: " << em << "/" << eM
303 << std::endl;
304}

307inline void
308initialize_parameters(const std::string &filename = "",
309 const std::string &output_filename = "")
310{
311 // Two-pass initialization:
312 // 1. Parse the file ignoring undeclared entries
313 // 2. From file to parameters
314 // 3. Declare additional acceptors
315 // 4. Parse again to create additional acceptors
316 // 5. From file to parameters
317 auto &prm = ParameterAcceptor::prm;
318 ParameterAcceptor::declare_all_parameters(prm);
319
320 if (!filename.empty())
321 {
322 try
323 {
324 prm.parse_input(filename, "", true);
325 ParameterAcceptor::parse_all_parameters(prm);
326
327 // Second pass. In this case, we do not skip undeclared entries,
328 // because all acceptors should have been created in the first pass,
329 // and we want to check that all entries in the file are valid.
330 ParameterAcceptor::declare_all_parameters(prm);
331 prm.parse_input(filename);
332 ParameterAcceptor::parse_all_parameters(prm);
333 }
334 catch (const ::ExcFileNotOpen &)
335 {
336 prm.print_parameters(filename, ParameterHandler::DefaultStyle);
337 AssertThrow(false,
338 ExcMessage("You specified <" + filename + "> as input " +
339 "parameter file, but it does not exist. " +
340 "We created it for you."));
341 }
342 }
343
344 if (!output_filename.empty())
345 prm.print_parameters(output_filename, ParameterHandler::Short);
346}

349inline void
350initialize_parameters_from_string(const std::string &prm_content,
351 const std::string &output_filename = "")
352{
353 // Two-pass initialization:
354 // 1) Parse the prm_content ignoring undeclared entries, so we can still read
355 // parameters that control the creation of additional acceptors
356 // (e.g., material tags).
357 // 2) Parse once to let acceptors create additional acceptors.
358 // 3) Parse again, now that the additional acceptors exist and have
359 // declared their parameters.
360 auto &prm = ParameterAcceptor::prm;
361 ParameterAcceptor::declare_all_parameters(prm);
362
363 prm.parse_input_from_string(prm_content, "", true);
364 ParameterAcceptor::parse_all_parameters(prm);
365
366 // Second pass.
367 ParameterAcceptor::declare_all_parameters(prm);
368 prm.parse_input_from_string(prm_content);
369 ParameterAcceptor::parse_all_parameters(prm);
370
371 if (!output_filename.empty())
372 prm.print_parameters(output_filename, ParameterHandler::Short);
373}
