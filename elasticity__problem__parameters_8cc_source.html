150 {
151 namespace fs = std::filesystem;
152 std::error_code ec;
153 fs::create_directories(output_directory, ec);
154 AssertThrow(!ec,
155 ExcMessage("Could not create output directory '" +
156 output_directory + "': " + ec.message()));
157 }
158
159 // Ensure material properties acceptors exist when material tags are
160 // provided. This is needed for the two-pass initialization strategy:
161 // - pass 1: read tags, create acceptors
162 // - pass 2: acceptors declare parameters and get parsed
163 if (material_properties_by_id.empty() &&
164 !material_tags_by_material_id.empty())
165 {
166 // Make sure we exit our subsection first, so that registration of the
167 // new classes works properly.
168 this->leave_my_subsection(this->prm);
169 for (const auto &[material_id, tag] : material_tags_by_material_id)
170 {
171 material_properties_by_id[material_id] =
172 std::make_unique<MaterialProperties>(tag);
173 }
174 // Go back to our subsection, so we can continue parsing parameters.
175 this->enter_my_subsection(this->prm);
176 // Do not check consistency yet: the dynamically created acceptors
177 // will only be parsed in the second parameter pass.
178 return;
179 }
180 else if (!material_tags_by_material_id.empty())
181 {
182 // Already initialized: check consistency.
183 AssertDimension(material_properties_by_id.size(),
184 material_tags_by_material_id.size());
185 // In the second pass, all material subsections have been parsed.
186 check_model_consistency();
187 return;
188 }
189
190 // No dynamic material tags: all information is already available.
191 check_model_consistency();
192 });
193}

196template <int dim, int spacedim>
197void
198ElasticityProblemParameters<dim, spacedim>::check_model_consistency()
199{
200 // --- Validate boundary id sets --------------------------------------------
201 const auto check_disjoint = [](const char *name_a,
202 const std::set<types::boundary_id> &a,
203 const char *name_b,
204 const std::set<types::boundary_id> &b) {
205 for (const auto id : a)
206 if (b.find(id) != b.end())
207 AssertThrow(false,
208 ExcMessage(std::string("Boundary id ") +
209 std::to_string(static_cast<unsigned int>(id)) +
210 " appears in both '" + name_a + "' and '" +
211 name_b + "'. These sets must be disjoint."));
212 };
213
214 check_disjoint("dirichlet ids",
215 dirichlet_ids,
216 "weak dirichlet ids",
217 weak_dirichlet_ids);
218 check_disjoint("dirichlet ids", dirichlet_ids, "neumann ids", neumann_ids);
219 check_disjoint("dirichlet ids",
220 dirichlet_ids,
221 "normal flux ids",
222 normal_flux_ids);
223 check_disjoint("weak dirichlet ids",
224 weak_dirichlet_ids,
225 "neumann ids",
226 neumann_ids);
227 check_disjoint("weak dirichlet ids",
228 weak_dirichlet_ids,
229 "normal flux ids",
230 normal_flux_ids);
231 check_disjoint("neumann ids",
232 neumann_ids,
233 "normal flux ids",
234 normal_flux_ids);
235
236 // Collect the set of materials that may be used: always include the default
237 // material, and also any explicitly configured material ids.
238 std::vector<std::reference_wrapper<const MaterialProperties>> materials;
239 materials.emplace_back(default_material_properties);
240 for (const auto &[id, mp_ptr] : material_properties_by_id)
241 {
242 (void)id;
243 Assert(mp_ptr != nullptr, ExcInternalError());
244 materials.emplace_back(*mp_ptr);
245 }
246
247 // --- Infer TimeMode -------------------------------------------------------
248 if (initial_time == final_time)
249 {
250 time_mode = TimeMode::Static;
251 }
252 else
253 {
254 bool any_rho_zero = false;
255 bool any_rho_positive = false;
256 for (const auto &mp_ref : materials)
257 {
258 const auto &mp = mp_ref.get();
259 AssertThrow(mp.rho >= 0.0,
260 ExcMessage("Material density (rho) must be >= 0."));
261 any_rho_zero |= (mp.rho == 0.0);
262 any_rho_positive |= (mp.rho > 0.0);
263 }
264
265 AssertThrow(!(any_rho_zero && any_rho_positive),
266 ExcMessage("Inconsistent densities: either all materials "
267 "must have rho == 0 (quasi-static) or all must "
268 "have rho > 0 (dynamic)."));
269
270 time_mode = any_rho_positive ? TimeMode::Dynamic : TimeMode::QuasiStatic;
271 }
272
273 // --- Validate BCs for time-dependent problems ---------------------------
274 if (time_mode != TimeMode::Static)
275 {
276 AssertThrow(dirichlet_ids.empty(),
277 ExcMessage("Time-dependent problems (initial_time != "
278 "final_time) do not support strong Dirichlet "
279 "boundary conditions. Use weak Dirichlet and/or "
280 "Neumann boundary conditions instead."));
281 AssertThrow(normal_flux_ids.empty(),
282 ExcMessage("Time-dependent problems (initial_time != "
283 "final_time) do not support normal-flux "
284 "constraints. Use weak Dirichlet and/or "
285 "Neumann boundary conditions instead."));
286 }
287
288 // --- Infer ElasticityModel ------------------------------------------------
289 bool any_neta_zero = false;
290 bool any_neta_positive = false;
291 for (const auto &mp_ref : materials)
292 {
293 const auto &mp = mp_ref.get();
294 AssertThrow(mp.neta >= 0.0,
295 ExcMessage("Material viscosity (eta/neta) must be >= 0."));
296 any_neta_zero |= (mp.neta == 0.0);
297 any_neta_positive |= (mp.neta > 0.0);
298 }
299
300 AssertThrow(!(any_neta_zero && any_neta_positive),
301 ExcMessage("Inconsistent viscosities: either all materials must "
302 "have eta == 0 or all must have eta > 0."));
303
304 if (any_neta_positive)
305 elasticity_model = ElasticityModel::KelvinVoigt;
306 else
307 elasticity_model = ElasticityModel::LinearElasticity;
308}

312inline const MaterialProperties &
313ElasticityProblemParameters<dim, spacedim>::get_material_properties(
314 const types::material_id material_id) const
315{
316 auto it = material_properties_by_id.find(material_id);

318 return *(it->second);
319 else
320 return default_material_properties;
321}
