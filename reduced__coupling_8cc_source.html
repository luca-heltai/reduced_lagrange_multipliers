88template <int reduced_dim, int dim, int spacedim, int n_components>
89void
90ReducedCoupling<reduced_dim, dim, spacedim, n_components>::initialize(
91 const Mapping<spacedim> &mapping)
92{
93 // Initialize the tensor product space
94 TensorProductSpace<reduced_dim, dim, spacedim, n_components>::initialize();
95
96 auto locally_owned_dofs = this->get_dof_handler().locally_owned_dofs();
97 auto locally_relevant_dofs =
98 DoFTools::extract_locally_relevant_dofs(this->get_dof_handler());
99
100 coupling_constraints.clear();
101 coupling_constraints.reinit(locally_owned_dofs, locally_relevant_dofs);
102 DoFTools::make_hanging_node_constraints(this->get_dof_handler(),
103 coupling_constraints);
104 coupling_constraints.close();
105
106 // Initialize the particle coupling
107 ParticleCoupling<spacedim>::initialize_particle_handler(
108 *this->background_tria, mapping);
109
110 // Initialize the particles
111 const auto &qpoints = this->get_locally_owned_qpoints();
112 const auto &weights = this->get_locally_owned_weights();
113 auto q_index = this->insert_points(qpoints, weights);
114 this->update_local_dof_indices(q_index);
115
116 // Initialize the coupling rhs
117 typename FunctionParser<spacedim>::ConstMap constants;
118 constants["pi"] = numbers::PI;
119 constants["E"] = numbers::E;
120
121 coupling_rhs = std::make_unique<FunctionParser<spacedim>>(
122 this->get_reference_cross_section().n_selected_basis());
123
124 coupling_rhs->initialize(FunctionParser<spacedim>::default_variable_names(),
125 par.coupling_rhs_expressions,
126 constants);
127
128 // This should be true. Let's double check
129 AssertDimension(coupling_rhs->n_components,
130 this->get_dof_handler().get_fe().n_components());
131
132 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
133 {
134 std::cout << "Reduced coupling initialized" << std::endl;
135 std::cout << "Reduced grid name: " << par.reduced_grid_name << std::endl;
136 }
137}

139template <int reduced_dim, int dim, int spacedim, int n_components>
140void
141ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
142 assemble_coupling_sparsity(DynamicSparsityPattern &dsp,
143 const DoFHandler<spacedim> &dh,
144 const AffineConstraints<double> &constraints) const
145{
146 const auto &fe = dh.get_fe();
147 std::vector<types::global_dof_index> background_dof_indices(
148 fe.n_dofs_per_cell());
149
150 auto particle = this->get_particles().begin();
151 while (particle != this->get_particles().end())
152 {
153 const auto &cell = particle->get_surrounding_cell();
154 const auto dh_cell =
155 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
156 dh_cell->get_dof_indices(background_dof_indices);
157
158 const auto pic = this->get_particles().particles_in_cell(cell);
159 Assert(pic.begin() == particle, ExcInternalError());
160
161 types::global_cell_index previous_cell_id = numbers::invalid_unsigned_int;
162 for (const auto &p : pic)
163 {
164 const auto [immersed_cell_id, immersed_q, section_q] =
165 this->particle_id_to_cell_and_qpoint_indices(p.get_id());
166 // If cell id is the same, we can skip the rest of the loop. We
167 // already added these entries
168 if (immersed_cell_id != previous_cell_id)
169 {
170 const auto &immersed_dof_indices =
171 this->get_dof_indices(immersed_cell_id);
172
173 constraints.add_entries_local_to_global(background_dof_indices,
174 coupling_constraints,
175 immersed_dof_indices,
176 dsp);
177
178 previous_cell_id = immersed_cell_id;
179 }
180 }
181 particle = pic.end();
182 }
183}
