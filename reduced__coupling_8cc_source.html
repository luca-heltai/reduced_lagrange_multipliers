71template <int reduced_dim, int dim, int spacedim, int n_components>
72void
73ReducedCoupling<reduced_dim, dim, spacedim, n_components>::initialize(
74 const Mapping<spacedim> &mapping)
75{
76 // Initialize the tensor product space
77 TensorProductSpace<reduced_dim, dim, spacedim, n_components>::initialize();
78
79 auto locally_owned_dofs = this->get_dof_handler().locally_owned_dofs();
80 auto locally_relevant_dofs =
81 DoFTools::extract_locally_relevant_dofs(this->get_dof_handler());
82
83 coupling_constraints.clear();
84 coupling_constraints.reinit(locally_owned_dofs, locally_relevant_dofs);
85 DoFTools::make_hanging_node_constraints(this->get_dof_handler(),
86 coupling_constraints);
87 coupling_constraints.close();
88
89 // Initialize the particle coupling
90 ParticleCoupling<spacedim>::initialize_particle_handler(
91 *this->background_tria, mapping);
92
93 // Initialize the particles
94 const auto &qpoints = this->get_locally_owned_qpoints();
95 const auto &weights = this->get_locally_owned_weights();
96 auto q_index = this->insert_points(qpoints, weights);
97 this->update_local_dof_indices(q_index);
98
99 // Initialize the coupling rhs
100 typename FunctionParser<spacedim>::ConstMap constants;
101 constants["pi"] = numbers::PI;
102 constants["E"] = numbers::E;
103
104 coupling_rhs = std::make_unique<FunctionParser<spacedim>>(
105 this->get_reference_cross_section().n_selected_basis());
106
107 coupling_rhs->initialize(FunctionParser<spacedim>::default_variable_names() +
108 ",t",
109 par.coupling_rhs_expressions,
110 constants,
111 true);
112
113 // This should be true. Let's double check
114 AssertDimension(coupling_rhs->n_components,
115 this->get_dof_handler().get_fe().n_components());
116
117 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
118 {
119 std::cout
120 << "Selected degree: "
121 << par.tensor_product_space_parameters.section.inclusion_degree
122 << ", selected basis functions: "
123 << Patterns::Tools::to_string(
124 par.tensor_product_space_parameters.section.selected_coefficients)
125 << std::endl;
126 std::cout << "Reduced coupling initialized" << std::endl;
127 std::cout << "Reduced grid name: "
128 << par.tensor_product_space_parameters.reduced_grid_name
129 << std::endl;
130 }
131}

133template <int reduced_dim, int dim, int spacedim, int n_components>
134void
135ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
136 assemble_coupling_sparsity(DynamicSparsityPattern &dsp,
137 const DoFHandler<spacedim> &dh,
138 const AffineConstraints<double> &constraints) const
139{
140 const auto &fe = dh.get_fe();
141 std::vector<types::global_dof_index> background_dof_indices(
142 fe.n_dofs_per_cell());
143
144 auto particle = this->get_particles().begin();
145 while (particle != this->get_particles().end())
146 {
147 const auto &cell = particle->get_surrounding_cell();
148 const auto dh_cell =
149 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
150 dh_cell->get_dof_indices(background_dof_indices);
151
152 const auto pic = this->get_particles().particles_in_cell(cell);
153 Assert(pic.begin() == particle, ExcInternalError());
154
155 types::global_cell_index previous_cell_id = numbers::invalid_unsigned_int;
156 for (const auto &p : pic)
157 {
158 const auto [immersed_cell_id, immersed_q, section_q] =
159 this->particle_id_to_cell_and_qpoint_indices(p.get_id());
160 // If cell id is the same, we can skip the rest of the loop. We
161 // already added these entries
162 if (immersed_cell_id != previous_cell_id)
163 {
164 const auto &immersed_dof_indices =
165 this->get_dof_indices(immersed_cell_id);
166
167 constraints.add_entries_local_to_global(background_dof_indices,
168 coupling_constraints,
169 immersed_dof_indices,
170 dsp);
171
172 previous_cell_id = immersed_cell_id;
173 }
174 }
175 particle = pic.end();
176 }
177}
