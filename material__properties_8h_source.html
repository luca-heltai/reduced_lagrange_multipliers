26struct MaterialProperties : public ParameterAcceptor
27{
28 MaterialProperties(const std::string &material_tag = "default")
29 : ParameterAcceptor("/Immersed Problem/Material properties/" +
30 material_tag + "/")
31 , material_tag(material_tag)
32 {
33 add_parameter("Lame mu", Lame_mu);
34 add_parameter("Lame lambda", Lame_lambda);
35 add_parameter("Density", rho);
36 add_parameter("Viscosity eta", neta);
37 add_parameter("Rayleigh alpha", rayleigh_alpha);
38 add_parameter("Rayleigh beta", rayleigh_beta);
39
40 parse_parameters_call_back.connect([this]() {
41 // Elastic modulus for isotropic materials in 3D
42 elastic_modulus =
43 Lame_mu * (3 * Lame_lambda + 2 * Lame_mu) / (Lame_lambda + Lame_mu);
44
45 // Poisson ratio for isotropic materials in 3D
46 poisson_ratio = Lame_lambda / (2 * (Lame_lambda + Lame_mu));
47
48 // Bulk modulus for isotropic materials in 3D
49 bulk_modulus = Lame_lambda + (2.0 / 3.0) * Lame_mu;
50
51 // Shear modulus is just Lame mu
52 shear_modulus = Lame_mu;
53 });
54 }
55
56 std::string material_tag = "default";
57 double Lame_mu = 1.0;
58 double Lame_lambda = 1.0;
59 double rho = 0.0;
60 double neta = 0.0;
61 double rayleigh_alpha = 0.0;
62 double rayleigh_beta = 0.0;
63 double elastic_modulus = 0.0;
64 double poisson_ratio = 0.0;
65 double bulk_modulus = 0.0;
66 double shear_modulus = 0.0;
67};
