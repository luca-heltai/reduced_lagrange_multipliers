165template <int reduced_dim, int dim, int spacedim, int n_components>
166void
167TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
168 update_local_dof_indices(
169 const std::map<unsigned int, IndexSet> &remote_q_point_indices)
170{
171 auto global_cell_indices =
172 local_q_point_indices_to_global_cell_indices(remote_q_point_indices);
173 std::map<
174 unsigned int,
175 std::map<types::global_cell_index, std::vector<types::global_dof_index>>>
176 global_dof_indices;
177
178 for (const auto &[proc, cell_indices] : global_cell_indices)
179 for (const auto &id : cell_indices)
180 global_dof_indices[proc][id] = global_cell_to_dof_indices[id];
181
182 // Exchange the data with participating processors
183 auto local_dof_indices =
184 Utilities::MPI::some_to_some(mpi_communicator, global_dof_indices);
185 // update global_cell_to_dof_indices
186 for (const auto &[proc, cell_indices] : local_dof_indices)
187 {
188 global_cell_to_dof_indices.insert(cell_indices.begin(),
189 cell_indices.end());
190 }
191}

206template <int reduced_dim, int dim, int spacedim, int n_components>
207std::tuple<unsigned int, unsigned int, unsigned int>
208TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
209 particle_id_to_cell_and_qpoint_indices(const unsigned int particle_id) const
210{
211 AssertIndexRange(particle_id,
212 triangulation.n_global_active_cells() *
213 quadrature_formula.size() *
214 reference_cross_section.n_quadrature_points());
215 const unsigned int cell_index =
216 particle_id /
217 (quadrature_formula.size() * reference_cross_section.n_quadrature_points());
218 const unsigned int qpoint_index_in_cell =
219 (particle_id / reference_cross_section.n_quadrature_points()) %
220 quadrature_formula.size();
221 const unsigned int qpoint_index_in_section =
222 particle_id % reference_cross_section.n_quadrature_points();
223
224 AssertIndexRange(cell_index, triangulation.n_global_active_cells());
225 AssertIndexRange(qpoint_index_in_cell, quadrature_formula.size());
226 AssertIndexRange(qpoint_index_in_section,
227 reference_cross_section.n_quadrature_points());
228 return std::make_tuple(cell_index,
229 qpoint_index_in_cell,
230 qpoint_index_in_section);
231}

234template <int reduced_dim, int dim, int spacedim, int n_components>
235std::map<unsigned int, IndexSet>
236TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
237 local_q_point_indices_to_global_cell_indices(
238 const std::map<unsigned int, IndexSet> &remote_q_point_indices) const
239{
240 std::map<unsigned int, IndexSet> cell_indices;
241 const IndexSet &owned_cells =
242 triangulation.global_active_cell_index_partitioner()
243 .lock()
244 ->locally_owned_range();
245
246 auto local_q_point_indices =
247 Utilities::MPI::some_to_some(mpi_communicator, remote_q_point_indices);
248
249 for (const auto &[proc, qpoint_indices] : local_q_point_indices)
250 {
251 IndexSet cell_indices_for_proc(triangulation.n_global_active_cells());
252 for (const auto &qpoint_index : qpoint_indices)
253 {
254 const auto [cell_index, q_index, i] =
255 particle_id_to_cell_and_qpoint_indices(qpoint_index);
256 cell_indices_for_proc.add_index(
257 owned_cells.nth_index_in_set(cell_index));
258 }
259 cell_indices_for_proc.compress();
260 cell_indices[proc] = cell_indices_for_proc;
261 }
262 return cell_indices;
263}

268void
269TensorProductSpace<reduced_dim, dim, spacedim, n_components>::setup_dofs()
270{
271 dof_handler.distribute_dofs(fe);
272 // Additional setup can be done here if needed
273 for (const auto &cell : dof_handler.active_cell_iterators())
274 if (cell->is_locally_owned())
275 {
276 std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
277 cell->get_dof_indices(dof_indices);
278 global_cell_to_dof_indices[cell->global_active_cell_index()] =
279 dof_indices;
280 }

283template <int reduced_dim, int dim, int spacedim, int n_components>
284IndexSet
285TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
286 locally_owned_qpoints() const
287{
288 IndexSet locally_owned_cell_set =
289 triangulation.global_active_cell_index_partitioner()
290 .lock()
291 ->locally_owned_range();
292
293 // Now make a tensor product of the local indices with the total number of
294 // quadrature points, and the number of quadrature points in the
295 // cross-section
296 const unsigned int n_qpoints_per_cell =
297 reference_cross_section.n_quadrature_points() * quadrature_formula.size();
298
299 IndexSet locally_owned_qpoints_set = locally_owned_cell_set.tensor_product(
300 complete_index_set(n_qpoints_per_cell));
301
302 return locally_owned_qpoints_set;
303}

327template <int reduced_dim, int dim, int spacedim, int n_components>
328void
329TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
330 compute_points_and_weights()
331{
332 all_qpoints.reserve(triangulation.n_active_cells() *
333 quadrature_formula.size() *
334 reference_cross_section.n_quadrature_points());
335 all_weights.reserve(triangulation.n_active_cells() *
336 quadrature_formula.size() *
337 reference_cross_section.n_quadrature_points());
338
339 reduced_qpoints.reserve(triangulation.n_active_cells() *
340 quadrature_formula.size());
341 reduced_weights.reserve(triangulation.n_active_cells() *
342 quadrature_formula.size());
343
344 UpdateFlags flags =
345 reduced_dim == 1 ?
346 update_quadrature_points | update_JxW_values :
347 update_quadrature_points | update_normal_vectors | update_JxW_values;
348
349 FEValues<reduced_dim, spacedim> fev(fe, quadrature_formula, flags);
350
351 for (const auto &cell : triangulation.active_cell_iterators())
352 if (cell->is_locally_owned())
353 {
354 fev.reinit(cell);
355 const auto &qpoints = fev.get_quadrature_points();
356 const auto &JxW = fev.get_JxW_values();
357
358 reduced_qpoints.insert(reduced_qpoints.end(),
359 qpoints.begin(),
360 qpoints.end());
361 for (const auto &w : JxW)
362 reduced_weights.emplace_back(std::vector<double>(1, w));
363
364 Tensor<1, spacedim> new_vertical;
365 if constexpr (reduced_dim == 1)
366 new_vertical = cell->vertex(1) - cell->vertex(0);
367
368 for (const auto &q : fev.quadrature_point_indices())
369 {
370 const auto &qpoint = qpoints[q];
371 if constexpr (reduced_dim == 2)
372 new_vertical = fev.normal_vector(q);
373 // [TODO] Make radius a function of the cell
374 auto cross_section_qpoints =
375 reference_cross_section.get_transformed_quadrature(qpoint,
376 new_vertical,
377 par.radius);
378 all_qpoints.insert(all_qpoints.end(),
379 cross_section_qpoints.get_points().begin(),
380 cross_section_qpoints.get_points().end());
381
382 for (const auto &w : cross_section_qpoints.get_weights())
383 all_weights.emplace_back(std::vector<double>(1, w * fev.JxW(q)));
384 }
385 }
386};

396template <int reduced_dim, int dim, int spacedim, int n_components>
397double
398TensorProductSpace<reduced_dim, dim, spacedim, n_components>::
399 weight_shape_value(const unsigned int i,
400 const unsigned int global_reduced_q,
401 const unsigned int section_q,
402 const unsigned int comp) const
403{
404 AssertIndexRange(i, reference_cross_section.n_selected_basis());
405 AssertIndexRange(section_q, reference_cross_section.n_quadrature_points());
406 AssertIndexRange(comp, n_components);
407 return reference_cross_section.shape_value(i, section_q, comp) *
408 get_scaling(global_reduced_q);
