111template <int dim, int spacedim>
112void
113read_grid_and_cad_files(const std::string &grid_file_name,
114 const std::string &ids_and_cad_file_names,
115 Triangulation<dim, spacedim> &tria)
116{
117 GridIn<dim, spacedim> grid_in;
118 grid_in.attach_triangulation(tria);
119 grid_in.read(grid_file_name);
120# ifdef DEAL_II_WITH_OPENCASCADE
121 using map_type = std::map<types::manifold_id, std::string>;
122 using Converter = Patterns::Tools::Convert<map_type>;
123 for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
124 {
125 const auto &manifold_id = pair.first;
126 const auto &cad_file_name = pair.second;
127 const auto extension = boost::to_lower_copy(
128 cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
129 TopoDS_Shape shape;
130 if (extension == "iges" || extension == "igs")
131 shape = OpenCASCADE::read_IGES(cad_file_name);
132 else if (extension == "step" || extension == "stp")
133 shape = OpenCASCADE::read_STEP(cad_file_name);
134 else
135 AssertThrow(false,
136 ExcNotImplemented("We found an extension that we "
137 "do not recognize as a CAD file "
138 "extension. Bailing out."));
139 const auto n_elements = OpenCASCADE::count_elements(shape);
140 if ((std::get<0>(n_elements) == 0))
141 tria.set_manifold(
142 manifold_id,
143 OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
144 else if (spacedim == 3)
145 {
146 const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
147 t->set_manifold(manifold_id,
148 OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
149 shape));
150 }
151 else
152 tria.set_manifold(manifold_id,
153 OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
154 TopoDS::Face(shape)));
155 }
156# else
157 (void)ids_and_cad_file_names;
158 AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
159# endif
160}

203# ifdef MATRIX_FREE_PATH

205# endif

209 relevant_dofs.resize(2);
210 DoFTools::extract_locally_relevant_dofs(dh, relevant_dofs[0]);
211 {
212 constraints.reinit(owned_dofs[0], relevant_dofs[0]);
213 DoFTools::make_hanging_node_constraints(dh, constraints);

216 constraints.close();
217 }
218 {
219# ifdef MATRIX_FREE_PATH
220 typename MatrixFree<spacedim, double>::AdditionalData additional_data;

234 mg_matrices.resize(0, nlevels - 1);
235
236 const std::set<types::boundary_id> dirichlet_boundary_ids = {0};
237 mg_constrained_dofs.initialize(dh);
238 mg_constrained_dofs.make_zero_boundary_constraints(
239 dh, dirichlet_boundary_ids);
240
241 for (unsigned int level = 0; level < nlevels; ++level)
242 {
243 AffineConstraints<double> level_constraints(
244 dh.locally_owned_mg_dofs(level),
245 DoFTools::extract_locally_relevant_level_dofs(dh, level));
246 for (const types::global_dof_index dof_index :
247 mg_constrained_dofs.get_boundary_indices(level))
248 level_constraints.constrain_dof_to_zero(dof_index);
249 level_constraints.close();
250
251 typename MatrixFree<spacedim, float>::AdditionalData
252 additional_data_level;
253 additional_data_level.tasks_parallel_scheme =
254 MatrixFree<spacedim, float>::AdditionalData::none;
255 additional_data_level.mapping_update_flags =
256 (update_gradients | update_JxW_values | update_quadrature_points);
257 additional_data_level.mg_level = level;
258 std::shared_ptr<MatrixFree<spacedim, float>> mg_mf_storage_level =
259 std::make_shared<MatrixFree<spacedim, float>>();
260 mg_mf_storage_level->reinit(mapping,
261 dh,
262 level_constraints,
263 QGauss<1>(fe->degree + 1),
264 additional_data_level);
265
266 mg_matrices[level].initialize(mg_mf_storage_level,
267 mg_constrained_dofs,
268 level);
269 }
270 }
271
272
273# else
274 stiffness_matrix.clear();
275 DynamicSparsityPattern dsp(relevant_dofs[0]);
276 DoFTools::make_sparsity_pattern(dh, dsp, constraints, false);
277 SparsityTools::distribute_sparsity_pattern(dsp,
278 owned_dofs[0],
279 mpi_communicator,
280 relevant_dofs[0]);
281 stiffness_matrix.reinit(owned_dofs[0],
282 owned_dofs[0],
283 dsp,
284 mpi_communicator);
285
286# endif
287 }
288 const auto &reduced_dh = reduced_coupling.get_dof_handler();
289 owned_dofs[1] = reduced_dh.locally_owned_dofs();
290 DoFTools::extract_locally_relevant_dofs(reduced_dh, relevant_dofs[1]);
291
292 coupling_matrix.clear();
293
294 DynamicSparsityPattern dsp(dh.n_dofs(),
295 reduced_dh.n_dofs(),
296 relevant_dofs[0]);
297
298 reduced_coupling.assemble_coupling_sparsity(dsp, dh, constraints);
299
300 coupling_matrix.reinit(owned_dofs[0], owned_dofs[1], dsp, mpi_communicator);
301
302 // DynamicSparsityPattern idsp(inclusions.n_dofs(),
303 // inclusions.n_dofs(),
304 // relevant_dofs[1]);
305 // for (const auto i : relevant_dofs[1])
306 // idsp.add(i, i);
307 // SparsityTools::distribute_sparsity_pattern(idsp,
308 // owned_dofs[1],
309 // mpi_communicator,

311 // inclusion_matrix.reinit(owned_dofs[1],
312 // owned_dofs[1],
313 // idsp,
314 // mpi_communicator);
315 // }
316
317 // Commented out inclusions-dependent reinit
318 locally_relevant_solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
319
320# ifdef MATRIX_FREE_PATH
321 system_rhs.reinit(owned_dofs, relevant_dofs, mpi_communicator);
322 solution.reinit(owned_dofs, relevant_dofs, mpi_communicator);
323# else
324 system_rhs.reinit(owned_dofs, mpi_communicator);

335# ifndef MATRIX_FREE_PATH
336template <int dim, int spacedim>
337void
338ReducedPoisson<dim, spacedim>::assemble_poisson_system()
339{
340 stiffness_matrix = 0;
341 coupling_matrix = 0;
342 system_rhs = 0;
343 FEValues<spacedim> fe_values(*fe,
344 *quadrature,
345 update_values | update_gradients |
346 update_quadrature_points | update_JxW_values);
347 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
348 const unsigned int n_q_points = quadrature->size();
349 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
350 Vector<double> cell_rhs(dofs_per_cell);
351 std::vector<double> rhs_values(n_q_points);
352 std::vector<Tensor<1, spacedim>> grad_phi_u(dofs_per_cell);
353 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
354 const FEValuesExtractors::Scalar scalar(0);
355 for (const auto &cell : dh.active_cell_iterators())
356 if (cell->is_locally_owned())
357 {
358 cell_matrix = 0;
359 cell_rhs = 0;
360 fe_values.reinit(cell);
361 par.rhs.value_list(fe_values.get_quadrature_points(), rhs_values);
362 for (unsigned int q = 0; q < n_q_points; ++q)
363 {
364 for (unsigned int k = 0; k < dofs_per_cell; ++k)
365 grad_phi_u[k] = fe_values[scalar].gradient(k, q);
366 for (unsigned int i = 0; i < dofs_per_cell; ++i)
367 {
368 for (unsigned int j = 0; j < dofs_per_cell; ++j)
369 {
370 cell_matrix(i, j) +=
371 grad_phi_u[i] * grad_phi_u[j] * fe_values.JxW(q);
372 }
373 cell_rhs(i) += fe_values.shape_value(i, q) * rhs_values[q] *
374 fe_values.JxW(q);
375 }
376 }
377 cell->get_dof_indices(local_dof_indices);
378 constraints.distribute_local_to_global(cell_matrix,
379 cell_rhs,
380 local_dof_indices,
381 stiffness_matrix,
382 system_rhs.block(0));
383 }
384 stiffness_matrix.compress(VectorOperation::add);
385 system_rhs.compress(VectorOperation::add);
386}

498template <int dim, int spacedim>
499void
500ReducedPoisson<dim, spacedim>::assemble_rhs()
501{
502 stiffness_matrix.get_matrix_free()->initialize_dof_vector(solution.block(0));
503 stiffness_matrix.get_matrix_free()->initialize_dof_vector(
504 system_rhs.block(0));
505 system_rhs = 0;
506 solution.block(0) = 0;
507 constraints.distribute(solution.block(0));
508 solution.block(0).update_ghost_values();
509
510 FEEvaluation<spacedim, -1> phi(*stiffness_matrix.get_matrix_free());
511 for (unsigned int cell = 0;
512 cell < stiffness_matrix.get_matrix_free()->n_cell_batches();
513 ++cell)
514 {
515 phi.reinit(cell);
516 phi.read_dof_values_plain(solution.block(0));
517 phi.evaluate(EvaluationFlags::gradients);
518 for (unsigned int q = 0; q < phi.n_q_points; ++q)
519 {
520 const Point<spacedim, VectorizedArray<double>> p_vect =
521 phi.quadrature_point(q);
522
523 VectorizedArray<double> f_value = 0.0;
524 for (unsigned int v = 0; v < VectorizedArray<double>::size(); ++v)
525 {
526 Point<spacedim> p;
527 for (unsigned int d = 0; d < spacedim; ++d)
528 p[d] = p_vect[d][v];
529 f_value[v] = par.rhs.value(p);
530 }
531 phi.submit_gradient(-phi.get_gradient(q), q);
532 phi.submit_value(f_value, q);
533 }
534 phi.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
535 phi.distribute_local_to_global(system_rhs.block(0));
536 }
537
538 system_rhs.compress(VectorOperation::add);

543template <int dim, int spacedim>
544void
545ReducedPoisson<dim, spacedim>::solve()
546{
547 TimerOutput::Scope t(computing_timer, "Solve");
548 pcout << "Preparing solve." << std::endl;
549 SolverCG<VectorType> cg_stiffness(par.inner_control);
550# ifdef MATRIX_FREE_PATH
551
552 using Payload = dealii::internal::LinearOperatorImplementation::EmptyPayload;
553 LinearOperator<VectorType, VectorType, Payload> A;
554 A = linear_operator<VectorType, VectorType, Payload>(stiffness_matrix);
555
556 MGTransferMatrixFree<spacedim, float> mg_transfer(mg_constrained_dofs);
557 mg_transfer.build(dh);
558
559 using SmootherType =
560 PreconditionChebyshev<LevelMatrixType,
561 LinearAlgebra::distributed::Vector<float>>;
562 mg::SmootherRelaxation<SmootherType,
563 LinearAlgebra::distributed::Vector<float>>
564 mg_smoother;
565 MGLevelObject<typename SmootherType::AdditionalData> smoother_data;
566 smoother_data.resize(0, tria.n_global_levels() - 1);
567 for (unsigned int level = 0; level < tria.n_global_levels(); ++level)
568 {
569 if (level > 0)
570 {
571 smoother_data[level].smoothing_range = 15.;
572 smoother_data[level].degree = 5;
573 smoother_data[level].eig_cg_n_iterations = 10;
574 }
575 else
576 {
577 smoother_data[0].smoothing_range = 1e-3;
578 smoother_data[0].degree = numbers::invalid_unsigned_int;
579 smoother_data[0].eig_cg_n_iterations = mg_matrices[0].m();
580 }
581 mg_matrices[level].compute_diagonal();
582 smoother_data[level].preconditioner =
583 mg_matrices[level].get_matrix_diagonal_inverse();
584 }
585 mg_smoother.initialize(mg_matrices, smoother_data);
586
587 MGCoarseGridApplySmoother<LinearAlgebra::distributed::Vector<float>>
588 mg_coarse;
589 mg_coarse.initialize(mg_smoother);
590
591 mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_matrix(mg_matrices);
592
593 MGLevelObject<MatrixFreeOperators::MGInterfaceOperator<LevelMatrixType>>
594 mg_interface_matrices;
595 mg_interface_matrices.resize(0, tria.n_global_levels() - 1);
596 for (unsigned int level = 0; level < tria.n_global_levels(); ++level)
597 mg_interface_matrices[level].initialize(mg_matrices[level]);
598 mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_interface(
599 mg_interface_matrices);
600
601 Multigrid<LinearAlgebra::distributed::Vector<float>> mg(
602 mg_matrix, mg_coarse, mg_transfer, mg_smoother, mg_smoother);
603 mg.set_edge_matrices(mg_interface, mg_interface);
604
605 PreconditionMG<spacedim,
606 LinearAlgebra::distributed::Vector<float>,
607 MGTransferMatrixFree<spacedim, float>>
608 preconditioner(dh, mg, mg_transfer);
609
610 auto invA = A;
611 invA = inverse_operator(A, cg_stiffness, preconditioner);
612# else
613 using Payload =
614 TrilinosWrappers::internal::LinearOperatorImplementation::TrilinosPayload;
615 LinearOperator<VectorType, VectorType, Payload> A;
616 A = linear_operator<VectorType, VectorType, Payload>(stiffness_matrix);
617
618 LA::MPI::PreconditionAMG prec_A;
619 {
620 LA::MPI::PreconditionAMG::AdditionalData data;
621# ifdef USE_PETSC_LA
622 data.symmetric_operator = true;
623# endif
624 pcout << "Initialize AMG...";
625 prec_A.initialize(stiffness_matrix, data);
626 pcout << "done." << std::endl;
627 }
628 const auto amgA = linear_operator<VectorType, VectorType, Payload>(A, prec_A);
629 auto invA = A;
630 invA = inverse_operator(A, cg_stiffness, amgA);
631# endif
632
633
634 // Some aliases
635 auto &u = solution.block(0);
636 auto &lambda = solution.block(1);
637
638 const auto &f = system_rhs.block(0);
639 const auto &g = system_rhs.block(1);
640

644 }

647# ifdef MATRIX_FREE_PATH

651 vec.reinit(owned_dofs[0], relevant_dofs[0], mpi_communicator);

657 const auto B = transpose_operator<VectorType, VectorType, Payload>(Bt);
658# else

662 // const auto B = linear_operator<VectorType, VectorType, Payload>(
663 // coupling_matrix_transpose);
664# endif

667 {
668 // Schur complement
669 pcout << " Prepare schur... ";
670 const auto S = B * invA * Bt;
671 pcout << "S was built." << std::endl;
672
673 // Schur complement preconditioner
674 auto invS = S;
675 SolverFGMRES<VectorType> solver_schur(par.outer_control);
676 invS =
677 inverse_operator<Payload, SolverFGMRES<VectorType>>(S,
678 solver_schur);
679
680 pcout << " f norm: " << f.l2_norm() << ", g norm: " << g.l2_norm()
681 << std::endl;
682
683 // Compute Lambda first
684 lambda = invS * (B * invA * f - g);
685 pcout << " Solved for lambda in " << par.outer_control.last_step()
686 << " iterations." << std::endl;
687
688 // Then compute u
689 u = invA * (f - Bt * lambda);
690 pcout << " u norm: " << u.l2_norm()
691 << ", lambda norm: " << lambda.l2_norm() << std::endl;
692
693 pcout << " Solved for u in " << par.inner_control.last_step()
694 << " iterations." << std::endl;
695
696 constraints.distribute(u);
697 reduced_coupling.get_coupling_constraints().distribute(lambda);
698 // solution.update_ghost_values();
699 locally_relevant_solution = solution;
700 }
701 else if (par.solver_name == "AL")
702 {
703 pcout << "Prepare AL preconditioner... " << std::endl;
704
705 AssertThrow(
706 (std::is_same_v<LA::MPI::Vector, TrilinosWrappers::MPI::Vector>),
707 ExcNotImplemented());
708
709 LA::MPI::SparseMatrix reduced_mass_matrix;
710 const auto &reduced_dh = reduced_coupling.get_dof_handler();
711 DynamicSparsityPattern dsp_reduced_mass(relevant_dofs[1]);
712 DoFTools::make_sparsity_pattern(reduced_dh, dsp_reduced_mass);
713 SparsityTools::distribute_sparsity_pattern(dsp_reduced_mass,
714 owned_dofs[1],
715 mpi_communicator,
716 relevant_dofs[1]);
717 reduced_mass_matrix.reinit(owned_dofs[1],
718 owned_dofs[1],

723 reduced_coupling.assemble_coupling_mass_matrix(reduced_mass_matrix);

726 << " x " << reduced_mass_matrix.n()
727 << ", norm: " << reduced_mass_matrix.linfty_norm() << std::endl;
728
729
730 const auto M = linear_operator<LA::MPI::Vector>(reduced_mass_matrix);
731
732 // Augmented Lagrangian solver
733 TrilinosWrappers::PreconditionILU M_inv_ilu;
734 M_inv_ilu.initialize(reduced_mass_matrix);
735
736 TrilinosWrappers::MPI::Vector inverse_squares_reduced; // diag(M)^{-2}
737 inverse_squares_reduced.reinit(owned_dofs[1], mpi_communicator);
738 for (const types::global_dof_index local_idx : owned_dofs[1])
739 {
740 const double el = reduced_mass_matrix.diag_element(local_idx);
741 Assert(std::abs(el) > 1e-10,
742 ExcMessage(
743 "Diagonal element " + std::to_string(local_idx) +
744 " of reduced mass matrix (" + std::to_string(el) +
745 ") is close to zero. Cannot compute inverse square."));
746 inverse_squares_reduced(local_idx) = 1. / (el * el);
747 }
748
749 inverse_squares_reduced.compress(VectorOperation::insert);

754 solver_control);

756 auto invW = invM * invM;
757
758 const double gamma = 10; // TODO: add to parameters file
759 auto Aug = A + gamma * Bt * invW * B;
760
761 TrilinosWrappers::SparseMatrix augmented_matrix;
762
763 TrilinosWrappers::PreconditionAMG prec_amg_augmented_block;
764 TrilinosWrappers::PreconditionAMG::AdditionalData data;

770 coupling_matrix,
771 inverse_squares_reduced,
772 gamma,
773 augmented_matrix);
774 prec_amg_augmented_block.initialize(augmented_matrix, data);
775 pcout << "done." << std::endl;
776 Aug = linear_operator<VectorType, VectorType, Payload>(
777 augmented_matrix);
778 }
779 else
780 {
781 prec_amg_augmented_block.initialize(stiffness_matrix, data);
782 }
783
784
785 auto Zero = M * 0.0;
786 auto AA = block_operator<2, 2, LA::MPI::BlockVector>(
787 {{{{Aug, Bt}}, {{B, Zero}}}});
788
789 LA::MPI::BlockVector solution_block;
790 LA::MPI::BlockVector system_rhs_block;
791 AA.reinit_domain_vector(solution_block, false);
792 AA.reinit_range_vector(system_rhs_block, false);
793
794 // lagrangian term
795 LA::MPI::Vector tmp;
796 tmp.reinit(system_rhs.block(0));
797 tmp = gamma * Bt * invW * system_rhs.block(1);
798 system_rhs_block.block(0) = system_rhs.block(0);
799 system_rhs_block.block(0).add(1., tmp); // ! augmented
800 system_rhs_block.block(1) = system_rhs.block(1);
801
802 SolverCG<LA::MPI::Vector> solver_lagrangian(par.inner_control);
803
804
805 auto Aug_inv =
806 inverse_operator(Aug, solver_lagrangian, prec_amg_augmented_block);
807
808 SolverFGMRES<LA::MPI::BlockVector> solver_fgmres(par.outer_control);
809 UtilitiesAL::BlockPreconditionerAugmentedLagrangian<LA::MPI::Vector>
810 augmented_lagrangian_preconditioner{Aug_inv, B, Bt, invW, gamma};
811
812 solver_fgmres.solve(AA,
813 solution_block,
814 system_rhs_block,
815 augmented_lagrangian_preconditioner);
816
817 pcout << " Solved with AL preconditioner in "
818 << par.outer_control.last_step() << " iterations." << std::endl;
819
820 constraints.distribute(solution_block.block(0));
821 reduced_coupling.get_coupling_constraints().distribute(
822 solution_block.block(1));
823 // solution.update_ghost_values();
824 locally_relevant_solution = solution_block;
825
826
827# ifdef DEBUG

829 {
