54template <int dim>
55double
56coefficient(const Point<dim> &p)
57{
58 if (p.square() < 0.5 * 0.5)
59 return 20;
60 else
61 return 1;
62}

74template <int dim>
75void
76Laplacian<dim>::setup_system()
77{
78 dof_handler.distribute_dofs(fe);
79
80 solution.reinit(dof_handler.n_dofs());
81 system_rhs.reinit(dof_handler.n_dofs());
82
83 constraints.clear();
84 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
85
86
87 VectorTools::interpolate_boundary_values(dof_handler,
88 0,
89 Functions::ZeroFunction<dim>(),
90 constraints);
91
92 constraints.close();
93
94 DynamicSparsityPattern dsp(dof_handler.n_dofs());
95 DoFTools::make_sparsity_pattern(dof_handler,
96 dsp,
97 constraints,
98 /*keep_constrained_dofs = */ false);
99
100 sparsity_pattern.copy_from(dsp);
101
102 system_matrix.reinit(sparsity_pattern);
103}

107template <int dim>
108void
109Laplacian<dim>::assemble_system()
110{
111 const QGauss<dim> quadrature_formula(fe.degree + 1);
112
113 FEValues<dim> fe_values(fe,
114 quadrature_formula,
115 update_values | update_gradients |
116 update_quadrature_points | update_JxW_values);
117
118 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
119
120 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
121 Vector<double> cell_rhs(dofs_per_cell);
122
123 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
124
125 for (const auto &cell : dof_handler.active_cell_iterators())
126 {
127 cell_matrix = 0;
128 cell_rhs = 0;
129
130 fe_values.reinit(cell);
131
132 for (const unsigned int q_index : fe_values.quadrature_point_indices())
133 {
134 const double current_coefficient =
135 coefficient<dim>(fe_values.quadrature_point(q_index));
136 for (const unsigned int i : fe_values.dof_indices())
137 {
138 for (const unsigned int j : fe_values.dof_indices())
139 cell_matrix(i, j) +=
140 (current_coefficient * // a(x_q)
141 fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
142 fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
143 fe_values.JxW(q_index)); // dx
144
145 cell_rhs(i) += (1.0 * // f(x)
146 fe_values.shape_value(i, q_index) * // phi_i(x_q)
147 fe_values.JxW(q_index)); // dx
148 }
149 }
150
151 cell->get_dof_indices(local_dof_indices);
152 constraints.distribute_local_to_global(
153 cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
154 }
155}

159template <int dim>
160void
161Laplacian<dim>::solve()
162{
163 SolverControl solver_control(1000, 1e-12);
164 SolverCG<Vector<double>> solver(solver_control);
165
166 PreconditionSSOR<SparseMatrix<double>> preconditioner;
167 preconditioner.initialize(system_matrix, 1.2);
168
169 solver.solve(system_matrix, solution, system_rhs, preconditioner);
170
171 constraints.distribute(solution);
172}

176template <int dim>
177void
178Laplacian<dim>::refine_grid()
179{
180 Vector<float> estimated_error_per_cell(triangulation.n_active_cells());
181
182 KellyErrorEstimator<dim>::estimate(dof_handler,
183 QGauss<dim - 1>(fe.degree + 1),
184 {},
185 solution,
186 estimated_error_per_cell);
187
188 GridRefinement::refine_and_coarsen_fixed_number(triangulation,
189 estimated_error_per_cell,
190 0.3,
191 0.03);
192
193 triangulation.execute_coarsening_and_refinement();
194}

198template <int dim>
199void
200Laplacian<dim>::output_results(const unsigned int cycle) const
201{
202 {
203 GridOut grid_out;
204 std::ofstream output("grid-" + std::to_string(cycle) + ".gnuplot");
205 GridOutFlags::Gnuplot gnuplot_flags(false, 5);
206 grid_out.set_flags(gnuplot_flags);
207 MappingQGeneric<dim> mapping(3);
208 grid_out.write_gnuplot(triangulation, output, &mapping);
209 }
210
211 {
212 DataOut<dim> data_out;
213 data_out.attach_dof_handler(dof_handler);
214 data_out.add_data_vector(solution, "solution");
215 data_out.build_patches();
216
217 std::ofstream output("solution-" + std::to_string(cycle) + ".vtu");
218 data_out.write_vtu(output);
219 }
220}

224template <int dim>
225void
226Laplacian<dim>::run()
227{
228 for (unsigned int cycle = 0; cycle < 8; ++cycle)
229 {
230 std::cout << "Cycle " << cycle << ':' << std::endl;
231
232 if (cycle == 0)
233 {
234 GridGenerator::hyper_ball(triangulation);
235 triangulation.refine_global(1);
236 }
237 else
238 refine_grid();
239
240
241 std::cout << " Number of active cells: "
242 << triangulation.n_active_cells() << std::endl;
243
244 setup_system();
245
246 std::cout << " Number of degrees of freedom: " << dof_handler.n_dofs()
247 << std::endl;
248
249 assemble_system();
250 solve();
251 output_results(cycle);
252 }
253}
