249template <int reduced_dim, int dim, int spacedim, int n_components>
250template <typename MatrixType>
251inline void
252ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
253 assemble_coupling_matrix(MatrixType &coupling_matrix,
254 const DoFHandler<spacedim> &dh,
255 const AffineConstraints<double> &constraints) const
256{
257 const auto &fe = dh.get_fe();
258 const auto &immersed_fe = this->get_dof_handler().get_fe();
259
260 std::vector<types::global_dof_index> background_dof_indices(
261 fe.n_dofs_per_cell());
262
263 FullMatrix<double> local_coupling_matrix(fe.n_dofs_per_cell(),
264 immersed_fe.n_dofs_per_cell());
265
266 FullMatrix<double> local_coupling_matrix_transpose(
267 immersed_fe.n_dofs_per_cell(), fe.n_dofs_per_cell());
268
269 auto particle = this->get_particles().begin();
270 while (particle != this->get_particles().end())
271 {
272 const auto &cell = particle->get_surrounding_cell();
273 const auto dh_cell =
274 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
275 dh_cell->get_dof_indices(background_dof_indices);
276
277 const auto pic = this->get_particles().particles_in_cell(cell);
278 Assert(pic.begin() == particle, ExcInternalError());
279
280 types::global_cell_index previous_cell_id = numbers::invalid_unsigned_int;
281 types::global_cell_index last_cell_id = numbers::invalid_unsigned_int;
282 local_coupling_matrix = 0;
283 for (const auto &p : pic)
284 {
285 const auto [immersed_cell_id, immersed_q, section_q] =
286 this->particle_id_to_cell_and_qpoint_indices(p.get_id());
287
288 const auto &background_p = p.get_reference_location();
289 const auto immersed_p = this->get_quadrature().point(immersed_q);
290 const auto &JxW = p.get_properties()[0];
291 last_cell_id = immersed_cell_id;
292 if (immersed_cell_id != previous_cell_id &&
293 previous_cell_id != numbers::invalid_unsigned_int)
294 {
295 // Distribute the matrix to the previous dofs
296 const auto &immersed_dof_indices =
297 this->get_dof_indices(previous_cell_id);
298 constraints.distribute_local_to_global(local_coupling_matrix,
299 background_dof_indices,
300 coupling_constraints,
301 immersed_dof_indices,
302 coupling_matrix);
303 local_coupling_matrix = 0;
304 previous_cell_id = immersed_cell_id;
305 }
306
307 for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
308 {
309 const auto comp_i = fe.system_to_component_index(i).first;
310 if (comp_i < n_components)
311 {
312 const auto v_i_comp_i = fe.shape_value(i, background_p);
313
314 for (unsigned int j = 0; j < immersed_fe.n_dofs_per_cell();
315 ++j)
316 {
317 const auto comp_j =
318 immersed_fe.system_to_component_index(j).first;
319
320 const auto phi_comp_j_comp_i =
321 this->get_reference_cross_section().shape_value(
322 comp_j, section_q, comp_i);
323
324 const auto w_j_comp_j =
325 immersed_fe.shape_value(j, immersed_p);
326
327 local_coupling_matrix(i, j) +=
328 v_i_comp_i * phi_comp_j_comp_i * w_j_comp_j * JxW;
329 }
330 }
331 }
332 }
333 const auto &immersed_dof_indices = this->get_dof_indices(last_cell_id);
334 constraints.distribute_local_to_global(local_coupling_matrix,
335 background_dof_indices,
336 coupling_constraints,
337 immersed_dof_indices,
338 coupling_matrix);
339 particle = pic.end();
340 }
341 coupling_matrix.compress(VectorOperation::add);
342}

344template <int reduced_dim, int dim, int spacedim, int n_components>
345template <typename MatrixType>
346inline void
347ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
348 assemble_coupling_mass_matrix(MatrixType &mass_matrix) const
349{
350 AssertDimension(mass_matrix.m(), this->get_dof_handler().n_dofs());
351 AssertDimension(mass_matrix.n(), this->get_dof_handler().n_dofs());
352
353 mass_matrix = 0;
354 const auto &fe = this->get_dof_handler().get_fe();
355
356 FullMatrix<double> local_mass_matrix(fe.n_dofs_per_cell(),
357 fe.n_dofs_per_cell());
358 std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
359 FEValues<reduced_dim, spacedim> fe_values(fe,
360 this->get_quadrature(),
361 update_values | update_JxW_values);
362
363
364 const auto &properties_fe = this->properties_dh.get_fe();
365 FEValues<reduced_dim, spacedim> properties_fe_values(properties_fe,
366 this->get_quadrature(),
367 update_values);
368
369 // Find the index of the thickness field in the properties
370 const unsigned int thickness_field_index =
371 std::distance(this->properties_names.begin(),
372 std::find(this->properties_names.begin(),
373 this->properties_names.end(),
374 par.thickness_field_name));
375
376 const auto thickness_start =
377 thickness_field_index >= this->properties_names.size() ?
378 numbers::invalid_unsigned_int :
379 VTKUtils::get_block_indices(properties_fe)
380 .block_start(thickness_field_index);
381
382 FEValuesExtractors::Scalar thickness(thickness_start);
383
384 // Initialize the thickness values with the constant thickness
385 std::vector<double> thickness_values(
386 this->get_quadrature().size(),
387 par.tensor_product_space_parameters.thickness);
388
389 for (const auto &cell : this->get_dof_handler().active_cell_iterators())
390 if (cell->is_locally_owned())
391 {
392 fe_values.reinit(cell);
393 const auto &JxW = fe_values.get_JxW_values();
394
395 if (thickness_start != numbers::invalid_unsigned_int)
396 {
397 properties_fe_values.reinit(
398 cell->as_dof_handler_iterator(this->properties_dh));
399 properties_fe_values[thickness].get_function_values(
400 this->properties, thickness_values);
401 }
402
403 local_mass_matrix = 0;
404 for (const auto q : fe_values.quadrature_point_indices())
405 {
406 const auto section_measure =
407 this->get_reference_cross_section().measure(thickness_values[q]);
408
409 for (const auto i : fe_values.dof_indices())
410 {
411 const auto comp_i =
412 fe_values.get_fe().system_to_component_index(i).first;
413 for (const auto j : fe_values.dof_indices())
414 {
415 const auto comp_j =
416 fe_values.get_fe().system_to_component_index(j).first;
417 if (comp_i == comp_j)
418 local_mass_matrix(i, j) += fe_values.shape_value(i, q) *
419 fe_values.shape_value(j, q) *
420 JxW[q] * section_measure;
421 }
422 }
423 }
424 cell->get_dof_indices(dof_indices);
425 coupling_constraints.distribute_local_to_global(local_mass_matrix,
426 dof_indices,
427 mass_matrix);
428 }
429 mass_matrix.compress(VectorOperation::add);
430}

442 update_values |

444 update_JxW_values);

451 this->get_dof_handler().get_fe().n_dofs_per_cell());
452
453
454 const auto &properties_fe = this->properties_dh.get_fe();
455 FEValues<reduced_dim, spacedim> properties_fe_values(properties_fe,
456 this->get_quadrature(),
457 update_values);

462 std::find(this->properties_names.begin(),

464 par.thickness_field_name));

467 thickness_field_index >= this->properties_names.size() ?
468 numbers::invalid_unsigned_int :
469 VTKUtils::get_block_indices(properties_fe)

471 ;
472

474
475 // Initialize the thickness values with the constant thickness
476 std::vector<double> thickness_values(
477 this->get_quadrature().size(),
478 par.tensor_product_space_parameters.thickness);
479
480 // VectorTools::create_right_hand_side(this->get_dof_handler(),
481 // this->get_quadrature(),
482 // *coupling_rhs,
483 // reduced_rhs,
484 // coupling_constraints);
485 for (const auto &cell : this->get_dof_handler().active_cell_iterators())
486 if (cell->is_locally_owned())
487 {
488 fe_values.reinit(cell);
489 const auto &JxW = fe_values.get_JxW_values();
490 const auto &q_points = fe_values.get_quadrature_points();
491 coupling_rhs->vector_value_list(q_points, rhs_values);
492
493
494 if (thickness_start != numbers::invalid_unsigned_int)

496 properties_fe_values.reinit(
497 cell->as_dof_handler_iterator(this->properties_dh));
498 properties_fe_values[thickness].get_function_values(
499 this->properties, thickness_values);

504 for (const auto i : fe_values.dof_indices())

507 fe_values.get_fe().system_to_component_index(i).first;
508 local_rhs(i) += rhs_values[q][comp_i] *
509 fe_values.shape_value(i, q) * JxW[q] *
510 this->get_reference_cross_section().measure(
511 thickness_values[q]);
512 }
513 cell->get_dof_indices(dof_indices);
514 coupling_constraints.distribute_local_to_global(local_rhs,
515 dof_indices,
