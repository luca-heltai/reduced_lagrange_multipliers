207template <int reduced_dim, int dim, int spacedim, int n_components>
208template <typename MatrixType>
209inline void
210ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
211 assemble_coupling_matrix(MatrixType &coupling_matrix,
212 const DoFHandler<spacedim> &dh,
213 const AffineConstraints<double> &constraints) const
214{
215 const auto &fe = dh.get_fe();
216 const auto &immersed_fe = this->get_dof_handler().get_fe();
217
218 std::vector<types::global_dof_index> background_dof_indices(
219 fe.n_dofs_per_cell());
220
221 FullMatrix<double> local_coupling_matrix(fe.n_dofs_per_cell(),
222 immersed_fe.n_dofs_per_cell());
223
224 FullMatrix<double> local_coupling_matrix_transpose(
225 immersed_fe.n_dofs_per_cell(), fe.n_dofs_per_cell());
226
227 auto particle = this->get_particles().begin();
228 while (particle != this->get_particles().end())
229 {
230 const auto &cell = particle->get_surrounding_cell();
231 const auto dh_cell =
232 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
233 dh_cell->get_dof_indices(background_dof_indices);
234
235 const auto pic = this->get_particles().particles_in_cell(cell);
236 Assert(pic.begin() == particle, ExcInternalError());
237
238 types::global_cell_index previous_cell_id = numbers::invalid_unsigned_int;
239 types::global_cell_index last_cell_id = numbers::invalid_unsigned_int;
240 local_coupling_matrix = 0;
241 for (const auto &p : pic)
242 {
243 const auto [immersed_cell_id, immersed_q, section_q] =
244 this->particle_id_to_cell_and_qpoint_indices(p.get_id());
245
246 // FIXME: this is not correct, we need to use the local
247 // immersed_cell_id index.
248 const auto global_reduced_qpoint_index = 0;
249 // immersed_cell_id * this->get_quadrature().size() + immersed_q;
250
251 const auto &background_p = p.get_reference_location();
252 const auto immersed_p = this->get_quadrature().point(immersed_q);
253 const auto &JxW = p.get_properties()[0];
254 last_cell_id = immersed_cell_id;
255 if (immersed_cell_id != previous_cell_id &&
256 previous_cell_id != numbers::invalid_unsigned_int)
257 {
258 // Distribute the matrix to the previous dofs
259 const auto &immersed_dof_indices =
260 this->get_dof_indices(previous_cell_id);
261 constraints.distribute_local_to_global(local_coupling_matrix,
262 background_dof_indices,
263 coupling_constraints,
264 immersed_dof_indices,
265 coupling_matrix);
266 local_coupling_matrix = 0;
267 previous_cell_id = immersed_cell_id;
268 }
269
270 for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
271 {
272 const auto comp_i = fe.system_to_component_index(i).first;
273 if (comp_i < n_components)
274 {
275 const auto v_i_comp_i = fe.shape_value(i, background_p);
276
277 for (unsigned int j = 0; j < immersed_fe.n_dofs_per_cell();
278 ++j)
279 {
280 const auto comp_j =
281 immersed_fe.system_to_component_index(j).first;
282
283 const auto phi_comp_j_comp_i = this->weight_shape_value(
284 comp_j, global_reduced_qpoint_index, section_q, comp_i);
285
286 const auto w_j_comp_j =
287 immersed_fe.shape_value(j, immersed_p);
288
289 local_coupling_matrix(i, j) +=
290 v_i_comp_i * phi_comp_j_comp_i * w_j_comp_j * JxW;
291 }
292 }
293 }
294 }
295 const auto &immersed_dof_indices = this->get_dof_indices(last_cell_id);
296 constraints.distribute_local_to_global(local_coupling_matrix,
297 background_dof_indices,
298 coupling_constraints,
299 immersed_dof_indices,
300 coupling_matrix);
301 particle = pic.end();
302 }
303 coupling_matrix.compress(VectorOperation::add);
304}
