234template <int reduced_dim, int dim, int spacedim, int n_components>
235template <typename MatrixType>
236inline void
237ReducedCoupling<reduced_dim, dim, spacedim, n_components>::
238 assemble_coupling_matrix(MatrixType &coupling_matrix,
239 const DoFHandler<spacedim> &dh,
240 const AffineConstraints<double> &constraints) const
241{
242 const auto &fe = dh.get_fe();
243 const auto &immersed_fe = this->get_dof_handler().get_fe();
244
245 std::vector<types::global_dof_index> background_dof_indices(
246 fe.n_dofs_per_cell());
247
248 FullMatrix<double> local_coupling_matrix(fe.n_dofs_per_cell(),
249 immersed_fe.n_dofs_per_cell());
250
251 FullMatrix<double> local_coupling_matrix_transpose(
252 immersed_fe.n_dofs_per_cell(), fe.n_dofs_per_cell());
253
254 auto particle = this->get_particles().begin();
255 while (particle != this->get_particles().end())
256 {
257 const auto &cell = particle->get_surrounding_cell();
258 const auto dh_cell =
259 typename DoFHandler<spacedim>::cell_iterator(*cell, &dh);
260 dh_cell->get_dof_indices(background_dof_indices);
261
262 const auto pic = this->get_particles().particles_in_cell(cell);
263 Assert(pic.begin() == particle, ExcInternalError());
264
265 types::global_cell_index previous_cell_id = numbers::invalid_unsigned_int;
266 types::global_cell_index last_cell_id = numbers::invalid_unsigned_int;
267 local_coupling_matrix = 0;
268 for (const auto &p : pic)
269 {
270 const auto [immersed_cell_id, immersed_q, section_q] =
271 this->particle_id_to_cell_and_qpoint_indices(p.get_id());
272
273 const auto &background_p = p.get_reference_location();
274 const auto immersed_p = this->get_quadrature().point(immersed_q);
275 const auto &JxW = p.get_properties()[0];
276 last_cell_id = immersed_cell_id;
277 if (immersed_cell_id != previous_cell_id &&
278 previous_cell_id != numbers::invalid_unsigned_int)
279 {
280 // Distribute the matrix to the previous dofs
281 const auto &immersed_dof_indices =
282 this->get_dof_indices(previous_cell_id);
283 constraints.distribute_local_to_global(local_coupling_matrix,
284 background_dof_indices,
285 coupling_constraints,
286 immersed_dof_indices,
287 coupling_matrix);
288 local_coupling_matrix = 0;
289 previous_cell_id = immersed_cell_id;
290 }
291
292 for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
293 {
294 const auto comp_i = fe.system_to_component_index(i).first;
295 if (comp_i < n_components)
296 {
297 const auto v_i_comp_i = fe.shape_value(i, background_p);
298
299 for (unsigned int j = 0; j < immersed_fe.n_dofs_per_cell();
300 ++j)
301 {
302 const auto comp_j =
303 immersed_fe.system_to_component_index(j).first;
304
305 const auto phi_comp_j_comp_i =
306 this->get_reference_cross_section().shape_value(
307 comp_j, section_q, comp_i);
308
309 const auto w_j_comp_j =
310 immersed_fe.shape_value(j, immersed_p);
311
312 local_coupling_matrix(i, j) +=
313 v_i_comp_i * phi_comp_j_comp_i * w_j_comp_j * JxW;
314 }
315 }
316 }
317 }
318 const auto &immersed_dof_indices = this->get_dof_indices(last_cell_id);
319 constraints.distribute_local_to_global(local_coupling_matrix,
320 background_dof_indices,
321 coupling_constraints,
322 immersed_dof_indices,
323 coupling_matrix);
324 particle = pic.end();
325 }
326 coupling_matrix.compress(VectorOperation::add);
327}

330template <int reduced_dim, int dim, int spacedim, int n_components>
331template <typename VectorType>
332inline void
333ReducedCoupling<reduced_dim, dim, spacedim, n_components>::assemble_reduced_rhs(
334 VectorType &reduced_rhs) const
335{
336 FEValues<reduced_dim, spacedim> fe_values(this->get_dof_handler().get_fe(),
337 this->get_quadrature(),
338 update_values |
339 update_quadrature_points |
340 update_JxW_values);
341
342 Vector<double> local_rhs(this->get_dof_handler().get_fe().n_dofs_per_cell());
343 std::vector<Vector<double>> rhs_values(
344 this->get_quadrature().size(),
345 Vector<double>(this->get_reference_cross_section().n_selected_basis()));
346 std::vector<types::global_dof_index> dof_indices(
347 this->get_dof_handler().get_fe().n_dofs_per_cell());
348
349 // VectorTools::create_right_hand_side(this->get_dof_handler(),
350 // this->get_quadrature(),
351 // *coupling_rhs,
352 // reduced_rhs,
353 // coupling_constraints);
354 for (const auto &cell : this->get_dof_handler().active_cell_iterators())
355 if (cell->is_locally_owned())
356 {
357 fe_values.reinit(cell);
358 const auto &JxW = fe_values.get_JxW_values();
359 const auto &q_points = fe_values.get_quadrature_points();
360 coupling_rhs->vector_value_list(q_points, rhs_values);
361
362 local_rhs = 0;
363 for (const auto q : fe_values.quadrature_point_indices())
364 for (const auto i : fe_values.dof_indices())
365 {
366 const auto comp_i =
367 fe_values.get_fe().system_to_component_index(i).first;
368 local_rhs(i) += rhs_values[q][comp_i] *
369 fe_values.shape_value(i, q) * JxW[q] *
370 this->get_reference_cross_section().measure(
371 par.tensor_product_space_parameters.radius);
372 }
373 cell->get_dof_indices(dof_indices);
374 coupling_constraints.distribute_local_to_global(local_rhs,
375 dof_indices,
376 reduced_rhs);
377 }
378
379 reduced_rhs.compress(VectorOperation::add);
380}
